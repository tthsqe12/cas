template <class T, class S, bool TA>
class BiElem {
public:
    T_elem_t exact;
    S_elem_t approx;
    bool have_exact;
    bool have_approx;

    BiElem() : have_exact(false), have_approx(false) {}
    BiElem(const S_elem_t& a) : approx(a), have_exact(false), have_approx(true) {}
    BiElem(const T_elem_t& a) : exact(a), have_exact(true), have_approx(false) {}

    void reset(const S_elem_t& a)
    {
        approx = a;
        have_exact = false;
        have_approx = true;
    }

    void reset(const T_elem_t& a)
    {
        exact = a;
        have_exact = true;
        have_approx = false;
    }
};

template <class S>
class BiElem<S, S, true> {
public:
    S_elem_t approx;

    BiElem() {}
    BiElem(const S_elem_t& a) : approx(a) {}

    void reset(const S_elem_t& a)
    {
        approx = a;
    }
};

template <class T, class S>
std::ostream& operator<<(std::ostream& o, const BiElem<T, S, false>& a)
{
    bool first = true;
    o << "BiElem[";
    if (a.have_exact)
    {
        o << a.exact;
        first = false;
    }
    if (a.have_approx)
    {
        if (first)
            o << ", ";
        o << a.approx;
        first = false;
    }
    o << "]";
    return o;
}

template <class S>
std::ostream& operator<<(std::ostream& o, const BiElem<S, S, true>& a)
{
    o << a.approx;
    return o;
}

template <class T, class S>
ex to_ex(const BiElem<T, S, false>& a)
{
    uex z; z.init_push_backx(emake_str("BiElem"), 2);
    if (a.have_exact)
        z.push_back(to_ex(a.exact));
    if (a.have_approx)
        z.push_back(to_ex(a.approx));
    return z.release();
}

template <class S>
ex to_ex(const BiElem<S, S, true>& a)
{
    uex z; z.init_push_backx(emake_str("BiElem"), 1);
    z.push_back(to_ex(a.approx));
    return z.release();
}




template <class T, class S, bool TA>
class BiRing {
public:
    T exact;
    S approx;

    typedef BiElem<T, S, false> elem_t;

    BiRing(T _exact, S _approx) : exact(_exact), approx(_approx) {}

    S_elem_t& approx_value(elem_t& a)
    {
        if (!a.have_approx)
        {
            a.have_approx = true;
            exact.nset(a.approx, a.exact, approx.precision());
        }
        return a.approx;
    }

    bool is_zero(const elem_t& a)
    {
        assert(a.have_exact || a.have_approx);
        if (a.have_exact)
            return exact.is_zero(a.exact);
        else
            return approx.is_zero(a.approx);
    }

    bool is_one(const elem_t& a)
    {
        assert(a.have_exact || a.have_approx);
        if (a.have_exact)
            return exact.is_one(a.exact);
        else
            return approx.is_one(a.approx);
    }

    bool is_equal(elem_t& a, elem_t& b)
    {
        assert(a.have_exact || a.have_approx);
        assert(b.have_exact || b.have_approx);
        if (a.have_exact && b.have_exact)
            return exact.is_equal(a.exact, b.exact);
        else
            return approx.is_equal(approx_value(a), approx_value(b));
    }

    bool is_integer(xfmpz_t& x, const elem_t& a)
    {
        assert(a.have_exact || a.have_approx);
        if (a.have_exact)
            return exact.is_integer(x, a.exact);
        else
            return approx.is_integer(x, a.approx);
    }

    void zero(elem_t& x)
    {
        exact.zero(x.exact);
        x.have_exact = true;
        x.have_approx = false;
    }

    void one(elem_t& x)
    {
        exact.one(x.exact);
        x.have_exact = true;
        x.have_approx = false;
    }

    void swap(elem_t& x, elem_t& y)
    {
        exact.swap(x.exact, y.exact);
        approx.swap(x.approx, y.approx);
        std::swap(x.have_exact, y.have_exact);
        std::swap(x.have_approx, y.have_approx);
    }

    void set(elem_t& x, const xfmpz_t& a)
    {
        exact.set(x.exact, a);
        x.have_exact = true;
        x.have_approx = false;
    }

    void set(elem_t& x, slong a)
    {
        exact.set(x.exact, a);
        x.have_exact = true;
        x.have_approx = false;
    }

    void set(elem_t& x, elem_t& a)
    {
        if (a.have_exact)
            exact.set(x.exact, a.exact);
        if (a.have_approx)
            approx.set(x.approx, a.approx);
        x.have_exact = a.have_exact;
        x.have_approx = a.have_approx;
    }

    void set(elem_t& x, const T_elem_t& a)
    {
        exact.set(x.exact, a);
        x.have_exact = true;
        x.have_approx = false;
    }

    void neg(elem_t& x, const elem_t& a)
    {
        if (a.have_exact)
            exact.neg(x.exact, a.exact);
        if (a.have_approx)
            approx.neg(x.approx, a.approx);
        x.have_exact = a.have_exact;
        x.have_approx = a.have_approx;
    }

    void neg(elem_t& x)
    {
        if (x.have_exact)
            exact.neg(x.exact);
        if (x.have_approx)
            approx.neg(x.approx);
    }

    void add(elem_t& x, elem_t& a, elem_t& b)
    {
        assert(a.have_exact || a.have_approx);
        assert(b.have_exact || b.have_approx);
        if (a.have_exact && b.have_exact)
        {
            exact.add(x.exact, a.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.add(x.approx, approx_value(a), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void add(elem_t& x, elem_t& a, const xfmpz_t& b)
    {
        if (a.have_exact)
        {
            exact.add(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.add(x.approx, approx_value(a), b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void add(elem_t& x, elem_t& a, slong b)
    {
        if (a.have_exact)
        {
            exact.add(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.add(x.approx, approx_value(a), b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void add(elem_t& x, elem_t& b)
    {
        assert(b.have_exact || b.have_approx);
        assert(x.have_exact || x.have_approx);
        if (x.have_exact && b.have_exact)
        {
            exact.add(x.exact, b.exact);
            x.have_approx = false;
        }
        else
        {
            approx.add(approx_value(x), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void add(elem_t& x, const xfmpz_t& b)
    {
        if (x.have_exact)
        {
            exact.add(x.exact, b);
            x.have_approx = false;
        }
        else
        {
            approx.add(x.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void add(elem_t& x, slong b)
    {
        if (x.have_exact)
        {
            exact.add(x.exact, b);
            x.have_approx = false;
        }
        else
        {
            approx.add(x.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void sub(elem_t& x, elem_t& a, elem_t& b)
    {
        assert(a.have_exact || a.have_approx);
        assert(b.have_exact || b.have_approx);
        if (a.have_exact && b.have_exact)
        {
            exact.sub(x.exact, a.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.sub(x.approx, approx_value(a), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void sub(elem_t& x, elem_t& a, slong b)
    {
        if (a.have_exact)
        {
            exact.sub(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.sub(x.approx, a.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void sub(elem_t& x, slong a, elem_t& b)
    {
        if (b.have_exact)
        {
            exact.sub(x.exact, a, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.sub(x.approx, a, approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void sub(elem_t& x, elem_t& b)
    {
        assert(b.have_exact || b.have_approx);
        assert(x.have_exact || x.have_approx);
        if (x.have_exact && b.have_exact)
        {
            exact.sub(x.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.sub(approx_value(x), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void sub(elem_t& x, slong b)
    {
        if (x.have_exact)
        {
            exact.sub(x.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.sub(x.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void mul(elem_t& x, elem_t& a, elem_t& b)
    {
        assert(a.have_exact || a.have_approx);
        assert(b.have_exact || b.have_approx);
        if (a.have_exact && b.have_exact)
        {
            exact.mul(x.exact, a.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.mul(x.approx, approx_value(a), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void mul(elem_t& x, elem_t& a, const xfmpz_t& b)
    {
        if (a.have_exact)
        {
            exact.mul(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.mul(x.approx, a.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void mul(elem_t& x, elem_t& a, slong b)
    {
        if (a.have_exact)
        {
            exact.mul(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.mul(x.approx, a.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void mul(elem_t& x, elem_t& b)
    {
        if (x.have_exact && b.have_exact)
        {
            exact.mul(x.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.mul(approx_value(x), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void mul(S_elem_t& x, elem_t& b)
    {
        if (b.have_exact)
            exact.nmul(x, b.exact, approx.precision());
        else
            approx.mul(x, b.approx);
    }

    void mul(elem_t& x, xfmpz_t& b)
    {
        if (x.have_exact)
        {
            exact.mul(x.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.mul(x.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void mul(elem_t& x, slong b)
    {
        if (x.have_exact)
        {
            exact.mul(x.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.mul(x.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void madd(elem_t& x, elem_t& a, elem_t& b)
    {
        assert(x.have_exact || x.have_approx);
        assert(a.have_exact || a.have_approx);
        assert(b.have_exact || b.have_approx);
        if (x.have_exact && a.have_exact && b.have_exact)
        {
            exact.madd(x.exact, a.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.madd(approx_value(x), approx_value(a), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void madd(elem_t& x, elem_t& a, xfmpz_t& b)
    {
        if (x.have_exact && a.have_exact)
        {
            exact.madd(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.madd(approx_value(x), approx_value(a), b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void msub(elem_t& x, elem_t& a, elem_t& b)
    {
        assert(x.have_exact || x.have_approx);
        assert(a.have_exact || a.have_approx);
        assert(b.have_exact || b.have_approx);
        if (x.have_exact && a.have_exact && b.have_exact)
        {
            exact.msub(x.exact, a.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.msub(approx_value(x), approx_value(a), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void msub(elem_t& x, elem_t& a, const xfmpz_t& b)
    {
        if (x.have_exact && a.have_exact)
        {
            exact.msub(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.msub(approx_value(x), approx_value(a), b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void inv(elem_t& x, elem_t&a)
    {
        if (a.have_exact)
        {
            exact.inv(x.exact, a.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.inv(x.approx, approx_value(a));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void divexact(elem_t& x, elem_t& a, elem_t& b)
    {
        assert(a.have_exact || a.have_approx);
        assert(b.have_exact || b.have_approx);
        if (a.have_exact && b.have_exact)
        {
            exact.divexact(x.exact, a.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.divexact(x.approx, approx_value(a), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void divexact(elem_t& x, elem_t& b)
    {
        assert(x.have_exact || x.have_approx);
        assert(b.have_exact || b.have_approx);
        if (x.have_exact && b.have_exact)
        {
            exact.divexact(x.exact, b.exact);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.divexact(approx_value(x), approx_value(b));
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void divexact(elem_t& x, elem_t& a, slong b)
    {
        assert(a.have_exact || a.have_approx);
        if (a.have_exact)
        {
            exact.divexact(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.divexact(x.approx, a.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }

    void pow(elem_t& x, elem_t& a, slong b)
    {
        assert(a.have_exact || a.have_approx);
        if (a.have_exact)
        {
            exact.pow(x.exact, a.exact, b);
            x.have_exact = true;
            x.have_approx = false;
        }
        else
        {
            approx.pow(x.approx, a.approx, b);
            x.have_exact = false;
            x.have_approx = true;
        }
    }
};

    void divexact(elem_t& x, elem_t& a, slong b)
    {
        approx.divexact(x.approx, a.approx, b);
    }

    void pow(elem_t& x, elem_t& a, slong b)
    {
        approx.pow(x.approx, a.approx, b);
    }
};


template <class T, class S, bool TA>
class BiFrac {
public:
    T_elem_t num;
    T_elem_t den;
    S_elem_t approx;
    bool have_exact;
    bool have_approx;

    BiFrac() : have_exact(false), have_approx(false) {}
    BiFrac(const S_elem_t& a) : approx(a), have_exact(false), have_approx(true) {}
    BiFrac(const T_elem_t& n, const T_elem_t& d) : num(n), den(d), have_exact(true), have_approx(false) {}

    void reset(const S_elem_t& a)
    {
        acb_set(approx.data, a);
        have_exact = false;
        have_approx = true;
    }

    void reset(const T_elem_t& a, const T_elem_t& b)
    {
        num = a;
        den = b;
        have_exact = true;
        have_approx = false;
    }

    S_elem_t& get_approx(BiRing<T, S, false>& R)
    {
        if (!have_approx)
        {
            S_elem_t n, d;
            R.nset(n, num, R.approx.precision());
            R.nset(d, den, R.approx.precision());
            R.approx.divexact(approx, n, d);
            have_approx = true;
        }
        return approx;
    }
};

// matrices of polynomials mod Lambda^tau
template <class T, class S, bool TA>
class BiMatrix {
public:
    std::vector<S_elem_t> approx;    
    std::vector<T_elem_t> num;
    T_elem_t den;
    bool have_exact;
    bool have_approx;

    BiMatrix() : have_exact(false), have_approx(false) {}

    BiMatrix(BiRing<T, S, false>& R, slong m, slong n, slong tau) :
        approx(m*n*tau), num(m*n*tau), have_exact(true), have_approx(false)
    {
        R.exact.one(den);
    }

    bool is_exact()
    {
        return have_exact;
    }

    void resize(slong n)
    {
        approx.resize(n);
        num.resize(n);
    }

    std::vector<S_elem_t>& approx_value(BiRing<T, S, false>& R)
    {
        if (!have_approx)
        {
            S_elem_t n, d;
            R.exact.nset(d, den, R.approx.precision());
            slong zn = num.size();
            assert(zn == approx.size());
            for (slong i = 0; i < zn; i++)
            {
                R.exact.nset(n, num[i], R.approx.precision());
                R.approx.divexact(approx[i], n, d);
            }
            have_approx = true;
        }
        return approx;
    }

    void force_approx(BiRing<T, S, false>& R)
    {
        if (!have_approx)
            approx_value(R);

        have_exact = false;
    }

    void limit_precision(BiRing<T, S, false>& R)
    {
        slong prec = approx.precision();

        if (!have_exact)
            return;

        if (R.nbits(den) > prec)
        {
            force_approx(R, prec);
            return;
        }

        for (const auto& zi : num)
        {
            if (R.nbits(zi) > prec)
            {
                force_approx(R, prec);
                return;
            }
        }
    }

};

    void resize(slong n)

        approx.resize(n);

    std::vector<S_elem_t>& approx_value(BiRing<S, S, false>& R)

        return approx;
    }

    void force_approx(BiRing<S, S, true>& R) {}
    void limit_precision(BiRing<S, S, true>& R) {}
};

template <>
hyp_maj<xarb_t>::get_series(arb_poly_t res, slong ord, slong prec)
{
    arb_poly_fit_length(res, ord);
    _arb_vec_zero(res->coeffs, ord);

    xarb_t t, t2;

    // z^i/(1-a*z^j)^k = Sum[Binomial[n+k-1,n]*a^n*z^(j*n+i),{n,0,Infinity}]
    xfmpz_t bin;
    for (auto& r : rats)
    {
        slong i = std::get<2>(r);
        slong j = std::get<3>(r);
        slong k = std::get<4>(r);
        fmpz_one(bin.data);
        for (slong n = 0; j*n + i < ord; n++)
        {
            arb_pow_ui(t.data, std::get<1>(r).data, n, prec);
            arb_mul_fmpz(t.data, t.data, bin.data);
            arb_addmul(res->coeffs + j*n + i, std::get<0>(r).data, t.data, prec);
            fmpz_mul_ui(bin.data, bin.data, n + k);
            fmpz_divexact_ui(bin.data, bin.data, n + 1);
        }
    }

    // Log[(1 + b*z^j)/(1 - a*z^j)] = Sum[(a^n-(-b)^n)/n*z^(j*n), {n,1,Infinity}]
    for (auto& r : rats)
    {
        slong j = std::get<3>(r);
        for (slong n = 1; j*n < ord; n++)
        {
            arb_pow_ui(t.data, std::get<1>(r).data, n, prec);
            arb_pow_ui(t2.data, std::get<2>(r).data, n, prec);
            if (n & 1)

            else
                arb_sub(t.data, t.data, t2.data);
            arb_div_ui(t.data, t.data, n);
            arb_addmul(res->coeffs + j*n, std::get<0>(r).data, t.data, prec);
        }
    }

    _arb_poly_set_length(res, ord);
    _arb_poly_normalise(res);

/** bs_product ****************************************************************/

template <class T, class E>
class bs_product {
public:
    slong top;
    E tmp;
    std::vector<std::tuple<E, slong, E>> stack;

    bs_product(T& R, E& x) : top(-1), tmp(x) {}

    bool is_empty()
    {
        return top < 0;
    }

    E& product()
    {
        return std::get<2>(stack.at(top));
    }

    void one(T& R)
    {
        top = -1;
    }

    E& push(T& R, E& x);
};

template <class T, class E>
E& bs_product<T, E>::push(T& R, E& x)
{
    slong n = top;
    auto & M = stack;
    n++;
    assert(n <= M.size());
    if (n >= M.size())
        M.emplace_back(tmp, 1, tmp);
    std::get<1>(M[n]) = 1;
    R.swap(std::get<0>(M[n]), x);

    while (n > 0 && std::get<1>(M[n]) >= std::get<1>(M[n-1]))
    {
        R.mul(tmp, std::get<0>(M[n]), std::get<0>(M[n-1]));
        R.swap(std::get<0>(M[n-1]), tmp);
        std::get<1>(M[n-1]) += std::get<1>(M[n]);
        n--;
    }

    if (n > 0)
        R.mul(std::get<2>(M[n]), std::get<0>(M[n]), std::get<2>(M[n-1]));
    else
        R.set(std::get<2>(M[n]), std::get<0>(M[n]));

    top = n;
    return std::get<2>(M[n]);
}

template <class E>
ex to_ex(const upoly<E>& a, er x)
{
    size_t n = a.coeffs.size();
    uex e; e.init_push_backr(gs.sym_sPlus.get(), n); 
    for (size_t i = n; i > 0; i--)
    {
        ex c = to_ex(a.coeffs[i - 1]);
        ex p = ex_powx(ecopy(x), emake_int_ui(i - 1));
        e.push_back(ex_mulx(c, p));
    }
    return ex_canonicalize_plus(e.release());
}

template <class E>
ex to_ex(const bpoly<E>& a, er x, er y)
{
    size_t n = a.coeffs.size();
    uex e; e.init_push_backr(gs.sym_sPlus.get(), n); 
    for (size_t i = n; i > 0; i--)
    {
        ex c = to_ex(a.coeffs[i - 1], x);
        ex p = ex_powx(ecopy(y), emake_int_ui(i - 1));
        e.push_back(emake_node(gs.sym_sNonCommutativeMultiply.copy(), c, p));
    }
    return ex_canonicalize_plus(e.release());
}

template <class T>
void normalize(
    T& R,
    upoly<T_elem_t>& a)
{
    size_t n = a.coeffs.size();
    if (n > 0 && R.is_zero(a.coeffs[n - 1]))
    {
        do {
            n--;
        } while (n > 0 && R.is_zero(a.coeffs[n - 1]));
        a.coeffs.resize(n);
    }
}

template <class T>
void normalize(
    T& R,
    bpoly<T_elem_t>& a)
{
    size_t n = a.coeffs.size();
    if (n > 0 && upoly_is_zero<T>(R, a.coeffs[n - 1]))
    {
        do {
            n--;
        } while (n > 0 && upoly_is_zero<T>(R, a.coeffs[n - 1]));
        a.coeffs.resize(n);
    }
}

template <class T>
void upoly_coeff(
    T& R,
    T_elem_t& c,
    upoly<T_elem_t>& a,
    slong i)
{
    if (i < 0 || i >= a.coeffs.size())
        R.zero(c);
    else
        R.set(c, a.coeffs[i]);
}

template <class T>
void upoly_add_term(
    T& R,
    upoly<T_elem_t>& x,
    size_t i,
    T_elem_t& c)
{
    size_t oldsize = x.coeffs.size();
    if (i >= oldsize)
    {
        x.coeffs.resize(i + 1);
        for (size_t j = oldsize; j <= i; j++)
            R.zero(x.coeffs[j]);
    }
    R.add(x.coeffs[i], c);
    normalize<T>(R, x);
}

template <class T>
void bpoly_add_term(
    T& R,
    bpoly<T_elem_t>& x,
    size_t i, // outside index
    size_t j, // inside index
    T_elem_t& c)
{
    if (i >= x.coeffs.size())
        x.coeffs.resize(i + 1);
    upoly_add_term<T>(R, x.coeffs[i], j, c);
    normalize<T>(R, x);
}

template <class T>
void bpoly_swap_vars(
    T& R,
    bpoly<T_elem_t>& x,
    bpoly<T_elem_t>& y)
{
    x.coeffs.clear();
    for (size_t i = y.coeffs.size(); i > 0; i--)
    {
        auto & yi = y.coeffs[i - 1];
        for (size_t j = yi.coeffs.size(); j > 0; j--)
            bpoly_add_term(R, x, j - 1, i - 1, yi.coeffs[j - 1]);
    }
}

template <class T>
void complete_xinside(
    T& R,
    BiBPoly<T_elem_t>& a)
{
    if (a.xinside.coeffs.empty())
        bpoly_swap_vars<T>(R, a.xinside, a.xoutside);
}

template <class T>
void complete_xoutside(
    T& R,
    BiBPoly<T_elem_t>& a)
{
    if (a.xoutside.coeffs.empty())
        bpoly_swap_vars<T>(R, a.xoutside, a.xinside);
}

template <class T>
void complete(
    T& R,
    BiBPoly<T_elem_t>& a)
{
    if (a.xinside.coeffs.empty())
        bpoly_swap_vars<T>(R, a.xinside, a.xoutside);
    else if (a.xoutside.coeffs.empty())
        bpoly_swap_vars<T>(R, a.xoutside, a.xinside);
}

template <class T>
bpoly<T_elem_t>& coeffs_xinside(
    T& R,
    LEquation<T_elem_t>& a)
{
    complete_xinside<T>(R, a.data);
    return a.data.xinside.coeffs;
}

template <class T>
bpoly<T_elem_t>& coeffs_xoutside(
    T& R,
    LEquation<T_elem_t>& a)
{
    complete_xoutside<T>(R, a.data);
    return a.data.xoutside;
}

template <class T>
bool upoly_is_equal(
    T& R,
    upoly<T_elem_t>& a,
    upoly<T_elem_t>& b)
{
    size_t n = a.coeffs.size();
    if (n != b.coeffs.size())
        return false;
    for (size_t i = 0; i < n; i++)
        if (!R.is_equal(a.coeffs[i], b.coeffs[i]))
            return false;
    return true;
}

    for (size_t i = 0; i < n; i++)
        if (!upoly_is_equal<T>(R, a.coeffs[i], b.coeffs[i]))
            return false;
    return true;
}

template <class T>
void upoly_add(
    T& R,
    upoly<T_elem_t>& x,
    upoly<T_elem_t>& a,
    upoly<T_elem_t>& b)
{
    assert(&x != &a && &x != &b);
    size_t an = a.coeffs.size();
    size_t bn = b.coeffs.size();
    size_t xn = std::max(an, bn);
    x.coeffs.resize(xn);
    for (size_t i = 0; i < xn; i++)
    {
        if (i >= an)            
            R.set(x.coeffs[i], b.coeffs[i]);
        else if (i >= bn)
            R.set(x.coeffs[i], a.coeffs[i]);
        else
            R.add(x.coeffs[i], a.coeffs[i], b.coeffs[i]);
    }
    normalize<T>(R, x);
}

template <class T>

}


template <class T>
void bpoly_add(
    T& R,
    bpoly<T_elem_t>& x,
    bpoly<T_elem_t>& a,
    bpoly<T_elem_t>& b)
{
    assert(&x != &a && &x != &b);
    size_t an = a.coeffs.size();
    size_t bn = b.coeffs.size();
    size_t xn = std::max(an, bn);
    x.coeffs.resize(xn);
    for (size_t i = 0; i < xn; i++)
    {
        if (i >= an)
            upoly_set<T>(R, x.coeffs[i], b.coeffs[i]);
        else if (i >= bn)
            upoly_set<T>(R, x.coeffs[i], a.coeffs[i]);
        else
            upoly_add<T>(R, x.coeffs[i], a.coeffs[i], b.coeffs[i]);
    }
    normalize<T>(R, x);
}

template <class T>
void bpoly_sub(
    T& R,
    bpoly<T_elem_t>& x,
    bpoly<T_elem_t>& a,
    bpoly<T_elem_t>& b)
{
    assert(&x != &a && &x != &b);
    size_t an = a.coeffs.size();
    size_t bn = b.coeffs.size();
    size_t xn = std::max(an, bn);
    x.coeffs.resize(xn);
    for (size_t i = 0; i < xn; i++)
    {
        if (i >= an)
            upoly_neg<T>(R, x.coeffs[i], b.coeffs[i]);
        else if (i >= bn)
            upoly_set<T>(R, x.coeffs[i], a.coeffs[i]);
        else
            upoly_sub<T>(R, x.coeffs[i], a.coeffs[i], b.coeffs[i]);
    }
    normalize<T>(R, x);
}

template <class T>
void upoly_evaluate(
    T& R,
    T_elem_t& z,
    upoly<T_elem_t>& a,
    T_elem_t& b)
{
    R.zero(z);
    for (size_t i = a.coeffs.size(); i > 0; i--)
    {
        R.mul(z, b);
        R.add(z, a.coeffs[i - 1]);
    }
}

template <class T>
void upoly_evaluate(
    T& R,
    T_elem_t& z,
    upoly<T_elem_t>& a,
    slong b)
{
    size_t i = a.coeffs.size();
    if (i > 1)
    {
        R.mul(z, a.coeffs[i - 1], b);
        R.add(z, a.coeffs[i - 2]);
        for (i -= 2; i > 0; i--)
        {
            R.mul(z, b);
            R.add(z, a.coeffs[i - 1]);
        }
    }
    else if (i == 1)
    {
        R.set(z, a.coeffs[0]);
    }
    else
    {
        R.zero(z);
    }
}

template <class T>
void upoly_mul_linear(
    T& R,
    upoly<T_elem_t>& a,
    T_elem_t& b)
{
    T_elem_t t, s;
    R.zero(s);
    for (auto & ai : a.coeffs)
    {
        R.mul(t, ai, b);
        R.add(s, t);
        R.swap(ai, s);
    }
    if (!R.is_zero(s))
        a.coeffs.push_back(std::move(s));
}

template <class T>
void upoly_mul_linear(
    T& R,
    upoly<T_elem_t>& a,
    T_elem_t& b,
    T_elem_t& c)
{
    T_elem_t s;
    R.zero(s);
    for (auto& ai : a.coeffs)
    {
        R.mul(s, s, c);
        R.madd(s, ai, b);
        R.swap(ai, s);
    }
    R.mul(s, s, c);
    if (!R.is_zero(s))
        a.coeffs.push_back(std::move(s));
}

template <class T>
void upoly_mul(
    T& R,
    upoly<T_elem_t>& x,
    upoly<T_elem_t>& a,
    upoly<T_elem_t>& b)
{
    assert(&x != &a && &x != &b);
    size_t an = a.coeffs.size();
    size_t bn = b.coeffs.size();
    size_t xn = an + bn - 1;
    if (an < 1 || bn < 1)
    {
        x.coeffs.clear();
        return;
    }
    x.coeffs.resize(xn);
    for (size_t i = 0; i < xn; i++)
    {
        // 0 <= j <= bn - 1
        // 0 <= i - j <= an - 1
        // i >= j >= -an+1 + i
        size_t jstop = std::min(i, bn - 1);
        size_t j = i + 1 >= an ? i + 1 - an : 0;        
        R.mul(x.coeffs.at(i), a.coeffs.at(i - j), b.coeffs.at(j));
        for (j++ ; j <= jstop; j++)
            R.madd(x.coeffs.at(i), a.coeffs.at(i - j), b.coeffs.at(j));
    }
    normalize<T>(R, x);
}

template <class T>
void upoly_divexact(
    T& R,
    upoly<T_elem_t>& q,
    upoly<T_elem_t>& A,
    upoly<T_elem_t>& b)
{
    assert(&q != &A && &q != &b);
    upoly<T_elem_t> a;
    upoly_set<T>(R, a, A);
    slong n = a.coeffs.size() - 1;
    slong m = b.coeffs.size() - 1;
    assert(n >= m && m > 0);
    q.coeffs.resize(n - m + 1);
    slong i = 0;
    slong j = n - m;
    slong l = 0;
    while (l < m && R.is_zero(b.coeffs.at(l)))
        l++;
    m -= l;
    n -= l;
    while (i <= j)
    {
        R.divexact(q.coeffs.at(i), a.coeffs.at(l + i), b.coeffs.at(l));
        for (slong k = 0; k <= m; k++)
            R.msub(a.coeffs.at(l + i + k), q.coeffs[i], b.coeffs[l + k]);
        if (++i > j)
            break;
        R.divexact(q.coeffs.at(j), a.coeffs.at(l + j + m), b.coeffs.at(l + m));
        for (slong k = 0; k <= m; k++)
            R.msub(a.coeffs.at(l + j + k), q.coeffs.at(j), b.coeffs.at(l + k));
        if (--j < i)
            break;
    }

    upoly_mul<T>(R, a, q, b);
    assert(upoly_is_equal<T>(R, a, A));
}

// R[x][y] * R[x]

template <class T>
void upoly_shift(
    T& R,
    upoly<T_elem_t>& z,
    slong n)
{
    if (n > 0)
    {
        size_t m = z.coeffs.size();
        z.coeffs.resize(m + n);
        for (size_t i = 0; i < n; i++)
            R.zero(z.coeffs.at(m + i));
        for (size_t i = m; i > 0; i--)
            R.swap(z.coeffs.at(i - 1), z.coeffs.at(i - 1 + n));
    }
    else if (n < 0)
    {
        n = -n;
        size_t m = z.coeffs.size();
        if (n >= m)
        {
            z.coeffs.clear();
            return;
        }

        for (size_t i = 0; i < m - n; i++)
            R.swap(z.coeffs.at(i), z.coeffs.at(i + n));

        z.coeffs.resize(m - n);
    }
}

template<class T>
void bpoly_shift_outside(
    T& R,
    bpoly<T_elem_t>& z,
    slong n)
{
    if (n > 0)
    {
        size_t m = z.coeffs.size();
        z.coeffs.resize(m + n);
        for (size_t i = 0; i < n; i++)
            z.coeffs.at(m + i).coeffs.clear();
        for (size_t i = m; i > 0; i--)
            std::swap(z.coeffs.at(i - 1), z.coeffs.at(i - 1 + n));
    }
    else
    {
        n = -n;
        size_t m = z.coeffs.size();
        if (n >= m)
        {
            z.coeffs.clear();
            return;
        }

        for (size_t i = 0; i < m - n; i++)
            std::swap(z.coeffs[i], z.coeffs[i + n]);

        z.coeffs.resize(m - n);
    }
}

    slong n)

template<class T>
void upoly_taylor_shift(
    T& R,
    upoly<T_elem_t>& x,
    upoly<T_elem_t>& a,
    T_elem_t& c)
{
    slong n = a.coeffs.size();
    x.coeffs.resize(n);
    if (n < 1)
        return;

    xfmpz_t s;
    std::vector<xfmpz_t> f(n, xfmpz_t(1));
    for (slong i = 0; i < n; i++)
    {
        R.zero(x.coeffs.at(i));
        for (slong j = n - 1; j >= i; j--)
        {
            R.mul(x.coeffs.at(i), c);
            R.madd(x.coeffs.at(i), a.coeffs.at(j), f.at(j));
        }
        fmpz_zero(s.data);
        for (slong j = i; j < n; j++)
        {
            fmpz_swap(f.at(j).data, s.data);
            fmpz_add(s.data, s.data, f.at(j).data);
        }
    }
}

template<class T>
void upoly_taylor_shift(
    T& R,
    upoly<T_elem_t>& x,
    upoly<T_elem_t>& a,
    slong c)
{
    slong n = a.coeffs.size();
    x.coeffs.resize(n);
    if (n < 1)
        return;

    xfmpz_t s;
    std::vector<xfmpz_t> f(n, xfmpz_t(1));
    for (slong i = 0; i < n; i++)
    {
        R.zero(x.coeffs.at(i));
        for (slong j = n - 1; j >= i; j--)
        {
            R.mul(x.coeffs.at(i), c);
            R.madd(x.coeffs.at(i), a.coeffs.at(j), f.at(j));
        }
        fmpz_zero(s.data);
        for (slong j = i; j < n; j++)
        {
            fmpz_swap(f.at(j).data, s.data);
            fmpz_add(s.data, s.data, f.at(j).data);
        }
    }
}

void stir2(fmpz_t z, slong n, slong i, slong m)
{
    assert(m == 0);
    arith_stirling_number_2(z, n, i);
}

template <class T>
void convert(
    T& R,
    DEquation<T_elem_t>& z,
    REquation<T_elem_t>& a)
{
    T_elem_t t;
    xfmpz_t f;
    z.data.xinside.coeffs.clear();
    z.data.xoutside.coeffs.clear();
    if (a.data.xinside.coeffs.empty())
    {
        auto & A = a.data.xoutside.coeffs;
        for (slong m = 0; m < A.size(); m++)
        {
            auto & Am = A[m].coeffs;
            for (slong n = 0; n < Am.size(); n++)
            for (slong i = 0; i <= n; i++)
            {
                stir2(f.data, n, i, 0);
                R.mul(t, Am[n], f);
                bpoly_add_term<T>(R, z.data.xinside, i, m + i, t);
            }
        }
    }
    else
    {
        auto & A = a.data.xinside.coeffs;
        for (slong n = 0; n < A.size(); n++)
        {
            auto & An = A[n].coeffs;
            for (slong m = 0; m < An.size(); m++)
            for (slong i = 0; i <= n; i++)
            {
                stir2(f.data, n, i, 0);
                R.mul(t, An[m], f);
                bpoly_add_term<T>(R, z.data.xinside, i, m + i, t);
            }
        }
    }
}

// input i is current degree
static void _mul_root(std::vector<xfmpz_t>& a, slong i, ulong r)
{
    fmpz_zero(a[i + 1].data);
    for (slong k = 0; k <= i; k++)
    {
        fmpz_submul_ui(a[i + 1].data, a[k].data, r);
        fmpz_swap(a[k].data, a[i + 1].data);
    }
}

template <class T>
void convert(
    T& R,
    REquation<T_elem_t>& z,
    DEquation<T_elem_t>& a)
{
    T_elem_t t;
    z.data.xinside.coeffs.clear();
    z.data.xoutside.coeffs.clear();
    complete_xinside<T>(R, a.data);
    auto & A = a.data.xinside.coeffs;
    std::vector<xfmpz_t> ppow(A.size() + 1);
    fmpz_one(ppow[0].data);
    for (slong i = 0; i < A.size(); i++)
    {
        auto & Ai = A[i].coeffs;
        for (slong j = 0; j < Ai.size(); j++)
        {
            if (j < i)
            {
                if (!R.is_zero(Ai[j]))
                    std::cout << "ignoring non-zero coefficient" << std::endl;
            }
            else
            {
                for (slong k = 0; k <= i; k++)
                {
                    R.mul(t, Ai[j], ppow[k].data);
                    bpoly_add_term<T>(R, z.data.xinside, k, j - i, t);
                }
            }
        }
        _mul_root(ppow, i, i);
    }
}

template <class T>
void convert(
    T& R,
    LEquation<T_elem_t>& z,
    DEquation<T_elem_t>& a)
{
    T_elem_t t;
    z.data.xinside.coeffs.clear();
    z.data.xoutside.coeffs.clear();
    complete_xinside<T>(R, a.data);
    auto & A = a.data.xinside.coeffs;
    std::vector<xfmpz_t> ppow(A.size() + 1);
    for (slong i = 0; i < A.size(); i++)
    {
        auto & Ai = A[i].coeffs;
        for (slong j = 0; j < Ai.size(); j++)
        {
            if (j < i)
            {
                if (!R.is_zero(Ai[j]))
                    std::cout << "ignoring non-zero coefficient" << std::endl;
            }
            else
            {
                fmpz_one(ppow[0].data);
                for (slong k = j - i; k < j; k++)
                    _mul_root(ppow, k - (j - i), k);

                for (slong k = 0; k <= i; k++)
                {
                    R.mul(t, Ai[j], ppow[k].data);
                    bpoly_add_term<T>(R, z.data.xinside, k, j - i, t);
                }
            }
        }
    }
}

/*
function transform_rescale(a::Union{LEquation{T}, REquation{T}}, s::T, F) where T
  A = a.data.xoutside.coeffs
  zxoutside = upoly{T}[upoly{T}([A[1+i].coeffs[1+j]*s^i
                       for j in 0:length(A[1+i].coeffs)-1])
                       for i in 0:length(A)-1]
  A = a.data.xinside.coeffs
  zxinside = upoly{T}[upoly{T}([A[1+i].coeffs[1+j]*s^j
                       for j in 0:length(A[1+i].coeffs)-1])
                       for i in 0:length(A)-1]
  return typeof(a)(BiBPoly{T}(bpoly{T}(zxinside), bpoly{T}(zxoutside)))
end
*/
template <class T>
void transform_rescale(
    T& R,
    LEquation<T_elem_t>& z,
    T_elem_t& s)
{
    T_elem_t t;

    auto & A = z.data.xoutside.coeffs;
    for (slong i = 0; i < A.size(); i++)
    for (slong j = 0; j < A[i].coeffs.size(); j++)
    {
        R.pow(t, s, i);
        R.mul(A[i].coeffs[j], t);
    }

    auto & B = z.data.xinside.coeffs;
    for (slong i = 0; i < B.size(); i++)
    for (slong j = 0; j < B[i].coeffs.size(); j++)
    {
        R.pow(t, s, j);
        R.mul(B[i].coeffs[j], t);
    }
}


template <class T>
void transform_shift(
    T& R,
    DEquation<T_elem_t>& z,
    DEquation<T_elem_t>& a,
    T_elem_t& c)
{
    complete_xinside<T>(R, a.data);
    auto & A = a.data.xinside.coeffs;
    auto & Z = z.data.xinside.coeffs;
    slong n = A.size();
    Z.resize(n);
    for (slong i = 0; i < n; i++)
        upoly_taylor_shift<T>(R, Z.at(i), A.at(i), c);
    z.data.xoutside.coeffs.clear();
}

/* equation for pFq(a;b|x) */

template <class T>
void pfq_equ_at_zero(
    T& R,
    LEquation<T_elem_t>& z,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    z.data.xinside.coeffs.clear();
    bpoly<T_elem_t>& Z = z.data.xoutside;
    T_elem_t t;

    Z.coeffs.resize(2);

    Z.coeffs[1].coeffs.resize(1);
    R.set(Z.coeffs[1].coeffs[0], -1);
    for (auto & ai : a)
    {
        R.sub(t, ai, 1);
        upoly_mul_linear<T>(R, Z.coeffs[1], t);
    }

    Z.coeffs[0].coeffs.resize(1);
    R.set(Z.coeffs[0].coeffs[0], 1);
    for (auto & bi : b)
    {
        R.sub(t, bi, 1);
        upoly_mul_linear<T>(R, Z.coeffs[0], t);
    }
    upoly_shift<T>(R, Z.coeffs[0], 1);
}

template <class T>
void pfq_equ_at_zero(
    T& R,
    REquation<T_elem_t>& z,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    z.data.xinside.coeffs.clear();
    bpoly<T_elem_t>& Z = z.data.xoutside;
    T_elem_t t;

    Z.coeffs.resize(2);

    Z.coeffs[1].coeffs.resize(1);
    R.set(Z.coeffs[1].coeffs[0], 1);
    for (auto & ai : a)
        upoly_mul_linear<T>(R, Z.coeffs[1], ai);

    Z.coeffs[0].coeffs.resize(1);
    R.set(Z.coeffs[0].coeffs[0], -1);
    for (auto & bi : b)
    {
        R.sub(t, bi, 1);
        upoly_mul_linear<T>(R, Z.coeffs[0], t);
    }
    upoly_shift<T>(R, Z.coeffs[0], 1);
}


/* equation for pFq(a;b|-1/x) */

template <class T>
void pfq_equ_at_infinity(
    T& R,
    LEquation<T_elem_t>& z,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    z.data.xinside.coeffs.clear();
    bpoly<T_elem_t>& Z = z.data.xoutside;
    T_elem_t t;

    Z.coeffs.resize(2);

    Z.coeffs[1].coeffs.resize(2);
    R.one(Z.coeffs[1].coeffs[0]);
    R.one(Z.coeffs[1].coeffs[1]);
    R.neg(Z.coeffs[1].coeffs[(a.size()^b.size()^1)&1]);
    for (auto & bi : b)
    {
        R.neg(t, bi);
        upoly_mul_linear<T>(R, Z.coeffs[1], t);
    }

    Z.coeffs[0].coeffs.resize(1);
    R.set(Z.coeffs[0].coeffs[0], 1);
    for (auto & ai : a)
    {
        R.neg(t, ai);
        upoly_mul_linear<T>(R, Z.coeffs[0], t);
    }
}

template <class T>
void pfq_equ_at_infinity(
    T& R,
    REquation<T_elem_t>& z,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    z.data.xinside.coeffs.clear();
    bpoly<T_elem_t>& Z = z.data.xoutside;
    T_elem_t t;

    Z.coeffs.resize(2);

    Z.coeffs[0].coeffs.resize(1);
    R.set(Z.coeffs[0].coeffs[0], ((a.size()^b.size())&1) ? 1 : -1);
    for (auto & ai : a)
    {
        R.neg(t, ai);
        upoly_mul_linear<T>(R, Z.coeffs[0], t);
    }

    Z.coeffs[1].coeffs.resize(1);
    R.set(Z.coeffs[1].coeffs[0], -1);
    for (auto & bi : b)
    {
        R.sub(t, 1, bi);
        upoly_mul_linear<T>(R, Z.coeffs[1], t);
    }
    upoly_shift<T>(R, Z.coeffs[1], 1);
}

/* equation for pFq(a;b|1-x) */
template <class T>
void pfq_equ_at_one(
    T& R,
    LEquation<T_elem_t>& z,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    REquation<T_elem_t> eR1;
    DEquation<T_elem_t> eD1, eD2;
    T_elem_t t;

    pfq_equ_at_zero<T>(R, eR1, a, b);
//std::cout << "eR1: " << eR1 << std::endl;
    convert<T>(R, eD1, eR1);
//std::cout << "eD1: " << eD1 << std::endl;
    shiftx<T>(R, eD1, -1);
//std::cout << "eD1: " << eD1 << std::endl;
    R.one(t);
    transform_shift<T>(R, eD2, eD1, t);
//std::cout << "eD2: " << eD2 << std::endl;
    shiftx<T>(R, eD2, b.size());
//std::cout << "eD2: " << eD2 << std::endl;
    convert<T>(R, z, eD2);
//std::cout << "z: " << z << std::endl;
    R.neg(t);
//std::cout << "t: " << t << std::endl;
    transform_rescale<T>(R, z, t);
//std::cout << "z: " << z << std::endl;
}


/* equation for x^((d-1)/2-σ)*exp(-d/x)*pFq(a;b|x^-d) */

template <class T>
void _at_inf_exp_step(
    T& R,
    bpoly<T_elem_t>& z,
    bpoly<T_elem_t>& a,
    slong b,
    T_elem_t c)
{
    slong n = a.coeffs.size() - 1;
    std::vector<T_elem_t> null;
    T_elem_t t;

    z.coeffs.resize(n + 2);
    for (slong i = 0; i <= n + 1; i++)
    {
        std::vector<T_elem_t>& t1 = (i > n)  ? null : a.coeffs.at(i).coeffs;
        std::vector<T_elem_t>& t2 = (i == 0) ? null : a.coeffs.at(i - 1).coeffs;
        slong m = std::max(t1.size(), t2.size());

        z.coeffs[i].resize(m + 1);
        std::vector<T_elem_t>& zi = z.coeffs[i].coeffs;
        for (slong j = 0; j <= m; j++)
        {
            if (j < t1.size())
                R.mul(zi[j], t1[j], b);
            else
                R.zero();

            if (0 <= j - 1 && j - 1 < t1.size())
            {
                R.sub(t, c, j - 1);
                R.madd(zi[j], t1[j - 1], t);
            }

            if (0 <= j - 1 && j - 1 < t1.size())
                R.sub(zi[j], t2[j - 1]);
        }
    }
}

template <class T>
void pfq_equ_at_infinity_exp(
    T& R,
    REquation<T_elem_t>& z,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    T_elem_t t;
    bpoly<T_elem_t> e1, e2, et;
    slong j;

    z.data.xoutside.coeffs.clear();
    slong p = a.size();
    slong q = b.size();
    slong d = q + 1 - p;
    assert(d > 0);

    T_elem_t sigma;
    R.set(sigma, 1 - d);
    R.mul_2exp(t, -1);
    for (auto& bi : b)
        R.add(sigma, bi);
    for (auto& ai : a)
        R.sub(sigma, ai);

    e1.coeffs.resize(1);
    e1.coeffs.at(0).coeffs.resize(1);
    R.set(t, d);
    R.pow(e1.coeffs.at(0).coeffs.at(0), t, d);
    j = 0;
    for (auto& ai : a)
    {
        R.sub(t, j, sigma);
        R.madd(t, ai, d);
        _at_inf_exp_step<T>(R, et, e1, d, t);
        std::swap(e1, et);
        j++;
    }

    e2.coeffs.resize(2);
    e2.coeffs.at(0).coeffs.resize(2);
    R.set(e2.coeffs.at(0).coeffs.at(0), -d);
    R.set(e2.coeffs.at(0).coeffs.at(1), sigma);
    e2.coeffs.at(1).coeffs.resize(2);
    R.zero(e2.coeffs.at(0).coeffs.at(0));
    R.one(e2.coeffs.at(0).coeffs.at(1));
    j = 0;
    for (auto& bi : b)
    {
        R.sub(t, j - d, sigma);
        R.madd(t, bi, d);
        _at_inf_exp_step<T>(R, et, e2, d, t);
        std::swap(e2, et);
        j++;
    }

    bpoly_add<T>(R, z.data.xoutside, e1, e2);
    bpoly_shift_inside<T>(R, z.data.xoutside, -1);
}


/* equation for (1+x)^(-2ai)*pFq(a;b|4x/(1+x)^2) */
/*
# return (c(x)*θ+b(x))*a
function _l_equ_step(a::bpoly{T}, c::upoly{T}, b::upoly{T}, F) where T
  # z_i = b*a_i + c*(x*a_i' + a_{i-1})
  A = a.coeffs
  n = length(A)-1
  z = upoly{T}[i<=n ? mul(b, A[1+i], F) : upoly{T}() for i in 0:n+1]
  for i in 0:n+1
    if i<1
      t = shift!(derivative(A[1+i], F), 1, F)
    elseif i>n
      t = A[1+i-1]
    else
      t = add(shift!(derivative(A[1+i], F), 1, F), A[1+i-1], F)
    end
    z[1+i] = add(z[1+i], mul(c, t, F), F)
  end
  return bpoly{T}(z)
end
*/

template <class T>
void _l_equ_step(
    T& R,
    bpoly<T_elem_t>& z,
    bpoly<T_elem_t>& a,
    upoly<T_elem_t>& c,
    upoly<T_elem_t>& b)
{
    upoly<T_elem_t> t, t2;
    slong m = a.coeffs.size();// n = m-1
    z.coeffs.resize(m + 1);
    for (slong i = 0; i <= m; i++)
    {
        if (i <= m-1)
            upoly_mul<T>(R, z.coeffs.at(i), a.coeffs.at(i), b);
        else
            upoly_zero<T>(R, z.coeffs.at(i));

        if (i < 1)
        {
            upoly_derivative<T>(R, t, a.coeffs.at(i));
            upoly_shift<T>(R, t, 1);
        }
        else if (i > m-1)
        {
            upoly_set<T>(R, t, a.coeffs.at(i - 1));
        }
        else
        {
            upoly_derivative<T>(R, t, a.coeffs.at(i));
            upoly_shift<T>(R, t, 1);
            upoly_add<T>(R, t, a.coeffs.at(i - 1));
        }

        upoly_mul<T>(R, t2, t, c);
        upoly_add<T>(R, z.coeffs.at(i), t2);
    }
}

template <class T>
void pfq_equ_at_zero_quadratic(
    T& R,
    REquation<T_elem_t>& z,
    slong k,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    upoly<T_elem_t> C, B;
    bpoly<T_elem_t> e1, e2, et;
    slong j;

    z.data.xoutside.coeffs.clear();
    C.coeffs.resize(3);
    B.coeffs.resize(3);
    R.set(C.coeffs.at(0), 1);
    R.set(C.coeffs.at(1), 0);
    R.set(C.coeffs.at(2), -1);

    e1.coeffs.resize(2);
    e1.coeffs.at(0).coeffs.resize(1);
    R.set(e1.coeffs.at(0).coeffs.at(0), a[k]);
    e1.coeffs.at(1).coeffs.resize(1);
    R.one(e1.coeffs.at(1).coeffs.at(0));
    j = 1;
    for (slong i = 0; i < a.size(); i++)
    {
        if (i == k)
            continue;
        R.set(B.coeffs[0], a[i]);
        R.sub(B.coeffs[1], a[k], a[i]);
        R.mul(B.coeffs[1], 2);
        R.add(B.coeffs[1], j + 1);
        R.mul(B.coeffs[2], a[k], -2);
        R.add(B.coeffs[2], a[i]);
        R.add(B.coeffs[2], j - 1);
        _l_equ_step<T>(R, et, e1, C, B);
        std::swap(e1, et);
        j += 2;
    }

    e2.coeffs.resize(2);
    e2.coeffs.at(0).coeffs.resize(2);
    R.zero(e2.coeffs.at(0).coeffs.at(0));
    R.mul(e2.coeffs.at(0).coeffs.at(1), a[k], 2);
    e2.coeffs.at(1).coeffs.resize(2);
    R.one(e2.coeffs.at(1).coeffs.at(0));
    R.one(e2.coeffs.at(1).coeffs.at(1));
    j = 1;
    for (auto & bi : b)
    {
        R.sub(B.coeffs.at(0), bi, 1);
        R.sub(B.coeffs.at(1), a[k], bi);
        R.mul(B.coeffs.at(1), 2);
        R.add(B.coeffs.at(1), j + 2);
        R.mul(B.coeffs.at(2), a[k], -2);
        R.add(B.coeffs.at(2), bi);
        R.add(B.coeffs.at(2), j - 1);
        _l_equ_step<T>(R, et, e2, C, B);
        std::swap(e2, et);
        j += 2;
    }

    C.coeffs.resize(2);
    R.one(C.coeffs.at(0));
    R.one(C.coeffs.at(1));
    bpoly_mul_inside<T>(R, et, e2, C);
    bpoly_mul_scalar<T>(R, e1, -4);
    bpoly_shift_inside<T>(R, e1, 1);
    bpoly_add<T>(R, et, e1);

    if (b.empty())
    {
        B.coeffs.resize(2);
        R.set(B.coeffs.at(1), -1);
    }
    else
    {
        B.coeffs.resize(3);
        R.set(B.coeffs.at(2), 1);
        R.set(B.coeffs.at(1), -2);
    }
    R.one(B.coeffs.at(0));
    bpoly_divexact_inside<T>(R, z.data.xinside, et, B);
}


template <class T>
void pfq_equ_at_zero_quadratic(
    T& R,
    LEquation<T_elem_t>& z,
    slong k,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b)
{
    REquation<T_elem_t> t;
    pfq_equ_at_zero_quadratic<T>(R, t, k, a, b);
    convert<T>(R, z, t);
}

template <class T>
void vec_add(
    T& R,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b,
    std::vector<T_elem_t>& c)
{
    size_t n = a.size();
    assert(n == b.size());
    assert(n == c.size());
    for (size_t i = 0; i < n; i++)
        R.add(a[i], b[i], c[i]);
}


template <class T>
void vec_mul_scalar(
    T& R,
    std::vector<T_elem_t>& x,
    T_elem_t& b)
{
    for (auto& xi : x)
        R.mul(xi, b);
}

template <class T>
void vec_mul_scalar(
    T& R,
    std::vector<T_elem_t>& x,
    std::vector<T_elem_t>& a,
    T_elem_t& b)
{
    size_t n = a.size();
    assert(x.size() == n);
    for (size_t i = 0; i < n; i++)
        R.mul(x[i], a[i], b);
}

template <class T>
void vec_madd_scalar(
    T& R,
    std::vector<T_elem_t>& a,
    std::vector<T_elem_t>& b,
    T_elem_t& c)
{
    size_t n = a.size();
    assert(b.size() == n);
    for (size_t i = 0; i < n; i++)
        R.madd(a[i], b[i], c);
}

template <class T, class S>
void mullow(
    BiRing<T, S, false>& R,
    BiMatrix<T, S, false>& z,  // m x n (x tau)
    BiMatrix<T, S, false>& a,  // m x p (x tau)
    BiMatrix<T, S, false>& b,  // p x n (x tau)
    slong m, slong p, slong n, slong tau)
{
    assert(m*n*tau == z.num.size());
    assert(m*n*tau == z.approx.size());
    assert(m*p*tau == a.num.size());
    assert(m*p*tau == a.approx.size());
    assert(p*n*tau == b.num.size());
    assert(p*n*tau == b.approx.size());

#define zIDX(i, j, k) [((i)*n + (j))*tau + (k)]
#define aIDX(i, j, k) [((i)*p + (j))*tau + (k)]
#define bIDX(i, j, k) [((i)*n + (j))*tau + (k)]

//std::cout << "mullow called" << std::endl;
//std::cout << "a.is_exact: " << a.is_exact << std::endl;
//std::cout << "b.is_exact: " << b.is_exact << std::endl;

    if (a.have_exact && b.have_exact)
    {
        z.have_exact = true;
        z.have_approx = false;
        R.exact.mul(z.den, a.den, b.den);

        for (slong i = 0; i < m; i++)
        for (slong j = 0; j < n; j++)
        for (slong k = 0; k < tau; k++)
            R.exact.zero(z.num zIDX(i,j,k));
        
        for (slong i = 0; i < m; i++)
        for (slong j = 0; j < n; j++)
        for (slong h = 0; h < p; h++)
        for (slong k = 0; k < tau; k++)
        for (slong l = 0; l <= k; l++)
            R.exact.madd(z.num zIDX(i,j,k), a.num aIDX(i,h,l), b.num bIDX(h,j,k-l));

        return;
    }

    z.have_exact = false;
    z.have_approx = true;
    for (slong i = 0; i < m; i++)
    for (slong j = 0; j < n; j++)
    for (slong k = 0; k < tau; k++)
        R.approx.zero(z.approx zIDX(i,j,k).data);

    if (!a.is_exact && !b.is_exact)
    {
        for (slong i = 0; i < m; i++)
        for (slong j = 0; j < n; j++)
        for (slong h = 0; h < p; h++)
        for (slong k = 0; k < tau; k++)
        for (slong l = 0; l <= k; l++)
            R.approx.madd(z.approx zIDX(i,j,k).data, a.approx aIDX(i,h,l).data, b.approx bIDX(h,j,k-l).data);
    }
    else
    {
        slong prec = R.approx.precision();
        S_elem_t t;

        assert(a.is_exact ^ b.is_exact);

        if (b.is_exact)
        {
            for (slong i = 0; i < m; i++)
            for (slong j = 0; j < n; j++)
            for (slong h = 0; h < p; h++)
            for (slong k = 0; k < tau; k++)
            for (slong l = 0; l <= k; l++)
                R.exact.nmadd(z.approx zIDX(i,j,k).data, a.approx aIDX(i,h,l).data, b.num bIDX(h,j,k-l), prec);

            R.exact.nset(t.data, b.den, prec);
        }
        else
        {
            for (slong i = 0; i < m; i++)
            for (slong j = 0; j < n; j++)
            for (slong h = 0; h < p; h++)
            for (slong k = 0; k < tau; k++)
            for (slong l = 0; l <= k; l++)
                R.exact.nmadd(z.approx zIDX(i,j,k).data, b.approx bIDX(h,j,k-l).data, a.num aIDX(i,h,l), prec);

            R.exact.nset(t.data, a.den, prec);
        }

        for (slong i = 0; i < m; i++)
        for (slong j = 0; j < n; j++)
        for (slong k = 0; k < tau; k++)
            R.approx.divexact(z.approx zIDX(i,j,k).data, z.approx zIDX(i,j,k).data, t.data);
    }

#undef zIDX
#undef aIDX
#undef bIDX
}

template <class S>
void mullow(
    BiRing<S, S, true>& R,
    BiMatrix<S, S, true>& z,  // m x n (x tau)
    BiMatrix<S, S, true>& a,  // m x p (x tau)
    BiMatrix<S, S, true>& b,  // p x n (x tau)
    slong m, slong p, slong n, slong tau)
{
    assert(m*n*tau == z.approx.size());
    assert(m*p*tau == a.approx.size());
    assert(p*n*tau == b.approx.size());

#define zIDX(i, j, k) [((i)*n + (j))*tau + (k)]
#define aIDX(i, j, k) [((i)*p + (j))*tau + (k)]
#define bIDX(i, j, k) [((i)*n + (j))*tau + (k)]

//std::cout << "approx mullow called" << std::endl;

    for (slong i = 0; i < m; i++)
    for (slong j = 0; j < n; j++)
    for (slong k = 0; k < tau; k++)
        R.approx.zero(z.approx zIDX(i,j,k).data);

    for (slong i = 0; i < m; i++)
    for (slong j = 0; j < n; j++)
    for (slong h = 0; h < p; h++)
    for (slong k = 0; k < tau; k++)
    for (slong l = 0; l <= k; l++)
        R.approx.madd(z.approx zIDX(i,j,k).data, a.approx aIDX(i,h,l).data, b.approx bIDX(h,j,k-l).data);

#undef zIDX
#undef aIDX
#undef bIDX
}

template <class T>
void numerator(
    T_int_ring_t& ZR,
    upoly<T_int_t>& x,
    T& R,
    upoly<T_elem_t>& a,
    T_elem_t& b)
{
    T_elem_t t;
    T_int_t d;

    size_t n = a.coeffs.size();
    x.coeffs.resize(n);
    for (size_t i = 0; i < n; i++)
    {
        R.divexact(t, a.coeffs[i], b);
        R.numerator_denominator(x.coeffs[i], d, t);
        assert(ZR.is_one(d));
    }
}


template <class T, class S, bool TA>
class SumBsCtx {
public:
    slong s, tau, delta;
    std::vector<upoly<T_elem_t>> eqnum;
    upoly<T_elem_t> eqden;
    BiFrac<T, S, TA> z;
    T_elem_t lambda_num;
    T_elem_t lambda_den;
    BiMatrix<T, S, TA> Sigma;
    BiMatrix<T, S, TA> M;
    BiMatrix<T, S, TA> Delta;
    BiMatrix<T, S, TA> Sigma0;
    BiMatrix<T, S, TA> M0;
    // temp space
    std::vector<std::pair<BiMatrix<T, S, TA>, BiMatrix<T, S, TA>>> tmatrix;
    std::vector<T_elem_t> tnumeval;
    T_elem_t tdeneval;
    // scratch space
    std::vector<T_elem_t> tff;
    T_elem_t t1, t2, t3, tzn, tzd;
    S_elem_t ta, ta1, ta2;

    std::vector<S_elem_t>& get_sum(BiRing<T, S, TA>& R)
    {
        return Sigma.get_approx(R);
    }

    std::pair<BiMatrix<T, S, TA>, BiMatrix<T, S, TA>> pop_tmatrix(BiRing<T, S, TA>& R)
    {
        if (tmatrix.empty())
        {
            return std::pair<BiMatrix<T, S, TA>, BiMatrix<T, S, TA>>(
                               BiMatrix<T, S, TA>(R, delta, s, tau),
                               BiMatrix<T, S, TA>(R, s, s, tau));
        }
        else
        {
            std::pair<BiMatrix<T, S, TA>, BiMatrix<T, S, TA>> res = std::move(tmatrix.back());
            tmatrix.pop_back();
            return res;
        }
    }

    void push_tmatrix(std::pair<BiMatrix<T, S, TA>, BiMatrix<T, S, TA>>&& a)
    {
        tmatrix.push_back(a);
    }
    
};

template <class T, class S, bool TA>
void sum_bs_der_recursive(
    BiRing<T, S, TA>& r,
    SumBsCtx<T, S, TA>& s,
    slong hi, slong lo);

//  in: sum [lo, mid) is in (S.Σ0, S.M0)
// out: put sum [lo, hi) in (S.Σ, S.M)
template <class T, class S, std::enable_if_t<!T::is_approx> = false>
void sum_bs_der_continue(
    BiRing<T, S, false>& r,
    SumBsCtx<T, S, false>& bs,
    slong hi, slong mid, slong lo)
{
    auto pt0 = bs.pop_tmatrix(r);
    BiMatrix<T, S, false>& Sigma0 = std::get<0>(pt0); 
    BiMatrix<T, S, false>& M0 = std::get<1>(pt0);
    std::swap(bs.Sigma0, Sigma0);
    std::swap(bs.M0, M0);

    assert(0 <= lo);
    assert(lo < mid);
    assert(mid < hi);

    sum_bs_der_recursive<T, S, false>(r, bs, hi, mid);

    auto pt1 = bs.pop_tmatrix(r);
    BiMatrix<T, S, false>& Sigma1 = std::get<0>(pt1); 
    BiMatrix<T, S, false>& M1 = std::get<1>(pt1);
    std::swap(bs.Sigma, Sigma1);
    std::swap(bs.M, M1);

//std::cout << "combining [" << lo << ", " << mid << ") with [" << mid << ", " << hi << ")"<< std::endl;
//std::cout << "Sigma1: " << Sigma1.num << "/" << Sigma1.den << std::endl;
//std::cout << "    M1: " << M1.num << "/" << M1.den << std::endl;
//std::cout << "Sigma0: " << Sigma0.num << "/" << Sigma0.den << std::endl;
//std::cout << "    M0: " << M0.num << "/" << M0.den << std::endl;

    mullow<T, S, false>(r, bs.Delta, Sigma1, M0, bs.delta, bs.s, bs.s, bs.tau);
    bs.Delta.limit_precision(r);
    if (bs.z.have_exact && bs.Delta.have_exact && Sigma0.have_exact)
    {
        r.exact.pow(bs.tzd, bs.z.den, mid - lo);
        r.exact.pow(bs.tzn, bs.z.num, mid - lo);
        bs.Sigma.is_exact = true;
        bs.Sigma.have_approx = false;
        r.exact.mul(bs.Sigma.den, bs.tzd, Sigma0.den);
        r.exact.mul(bs.Sigma.den, Sigma1.den);
        r.exact.divexact(bs.t2, Sigma0.den, M0.den);
        r.exact.mul(bs.t3, bs.t2, bs.tzn);
        mul_scalar(r.exact, bs.Sigma.num, bs.Delta.num, bs.t3);
        r.exact.mul(bs.t3, Sigma1.den, bs.tzd);
        madd_scalar(r.exact, bs.Sigma.num, Sigma0.num, bs.t3);
        bs.Sigma.limit_precision(r);
        r.exact.mul(bs.Delta.den, bs.tzd);
        mul_scalar(r, bs.Delta.num, bs.tzn);
    }
    else
    {
        bs.Delta.force_approx(r);
        r.approx.pow(bs.ta, bs.z.approx_value(r), mid - lo);
        mul_scalar(r.approx, bs.Delta.approx, bs.ta);
        bs.Sigma.is_exact = false;
        bs.Sigma.have_approx = true;
        add(bs.Sigma.approx, Sigma0.approx_value(r), bs.Delta.approx);
    }
    mullow<T, S, false>(r, bs.M, M1, M0, bs.s, bs.s, bs.s, bs.tau);
    bs.M.limit_precision(r);

//std::cout << "bs.Sigma: " << bs.Sigma.num << "/" << bs.Sigma.den << std::endl;
//std::cout << "    bs.M: " << bs.M.num << "/" << bs.M.den << std::endl;

    bs.push_tmatrix(std::move(pt1));

    std::swap(bs.Sigma0, Sigma0);
    std::swap(bs.M0, M0);
    bs.push_tmatrix(std::move(pt0));
}


template <class T, class S, std::enable_if_t<T::is_approx, bool> = true>
void sum_bs_der_continue(
    BiRing<S, S, true>& r,
    SumBsCtx<S, S, true>& bs,
    slong hi, slong mid, slong lo)
{
    auto pt0 = bs.pop_tmatrix(r);
    BiMatrix<S, S, true>& Sigma0 = std::get<0>(pt0); 
    BiMatrix<S, S, true>& M0 = std::get<1>(pt0);
    std::swap(bs.Sigma0, Sigma0);
    std::swap(bs.M0, M0);

    assert(0 <= lo);
    assert(lo < mid);
    assert(mid < hi);

    sum_bs_der_recursive<S, S, true>(r, bs, hi, mid);

    auto pt1 = bs.pop_tmatrix(r);
    BiMatrix<S, S, true>& Sigma1 = std::get<0>(pt1); 
    BiMatrix<S, S, true>& M1 = std::get<1>(pt1);
    std::swap(bs.Sigma, Sigma1);
    std::swap(bs.M, M1);

    mullow<S, S, true>(r, bs.Delta, Sigma1, M0, bs.delta, bs.s, bs.s, bs.tau);
    r.approx.pow(bs.ta, bs.z.approx, mid - lo);
    vec_mul_scalar(r.approx, bs.Delta.approx, bs.ta);
    add(bs.Sigma.approx, Sigma0.approx_value(r), bs.Delta.approx);
    mullow<S, S, false>(r, bs.M, M1, M0, bs.s, bs.s, bs.s, bs.tau);

    bs.push_tmatrix(std::move(pt1));

    std::swap(bs.Sigma0, Sigma0);
    std::swap(bs.M0, M0);
    bs.push_tmatrix(std::move(pt0));
}



// out: put sum from [lo, hi) in (S.Σ, S.M)
template <class T, class S, std::enable_if_t<!T::is_approx, bool> = false>
void sum_bs_der_recursive(
    BiRing<T, S, false>& r,
    SumBsCtx<T, S, false>& bs,
    slong hi, slong lo)
{
    assert(hi > lo);

//std::cout << "sum_bs_der_recursive called" << std::endl;
//std::cout << "hi: " << hi << std::endl;
//std::cout << "lo: " << lo << std::endl;

    assert(0 <= lo);
    assert(lo < hi);

    if (hi > lo + (bs.z.is_exact ? 1 : 1))
    {
        slong mid = (hi + lo)/2;
        sum_bs_der_recursive<T, S, false>(r, bs, mid, lo);
        std::swap(bs.Sigma0, bs.Sigma);
        std::swap(bs.M0, bs.M);
        sum_bs_der_continue<T, S, false>(r, bs, hi, mid, lo);
        return;
    }

    slong delta = bs.delta;
    slong s = bs.s;
    slong tau = bs.tau;

#define IDX(i0,i1,i2) [((i0)*(s) + (i1))*(tau) + (i2)]

//    bs.Sigma.resize(delta*s*tau);
//    bs.M.resize(s*s*tau);
//    bs.Delta.resize(delta*s*tau);

//std::cout << "bs.eqnum: " << bs.eqnum << std::endl;
//std::cout << "bs.eqden: " << bs.eqden << std::endl;

    bs.Sigma.is_exact = true;
    bs.Sigma.have_approx = false;
    r.exact.one(bs.Sigma.den);
    for (slong d = 0; d < delta; d++)
    for (slong j = 0; j < s; j++)
    for (slong k = 0; k < tau; k++)
        r.exact.zero(bs.Sigma.num IDX(d,j,k));

    bs.M.is_exact = true;
    bs.M.have_approx = false;
    r.exact.one(bs.M.den);
    for (slong i = 0; i < s; i++)
    for (slong j = 0; j < s; j++)
    for (slong k = 0; k < tau; k++)
        r.exact.zero(bs.M.num IDX(i,j,k));

    upoly_evaluate<T>(r.exact, bs.M.den, bs.eqden, lo);
    for (slong j = 0; j < s; j++)
    for (slong k = 0; k < tau; k++)
        upoly_evaluate<T>(r.exact, bs.M.num IDX(0,j,k), bs.eqnum[j*tau+k], lo);

    for (slong i = 1; i < s; i++)
        r.exact.set(bs.M.num IDX(i,i-1,0), bs.M.den);

    // ff = λden^(δ-1)*(Λ+λ+n-0)*(Λ+λ+n-1)*...*(Λ+λ+n-(d-1))
    slong fflen = 1;
    r.exact.pow(bs.tff[0], bs.lambda_den, delta - 1);

    r.exact.mul(bs.Sigma.den, bs.tff[0], bs.M.den);
    for (slong j = 0; j < s; j++)
    for (slong k = 0; k < tau; k++)
        r.exact.mul(bs.Sigma.num IDX(0,j,k), bs.tff[0], bs.M.num IDX(0,j,k));

    for (slong d = 1; d < delta; d++)
    {
        r.exact.mul(bs.t1, bs.lambda_den, lo - (d - 1));
        r.exact.add(bs.t1, bs.lambda_num);

        // ff *= lo-(d-1)+λ + Λ
        for (slong i = 0; i < fflen; i++)
            r.exact.divexact(bs.tff[i], bs.lambda_den);

        for (slong i = fflen - 1; i > 0; i--)
        {
            r.exact.mul(bs.tff[i], bs.t1);
            r.exact.madd(bs.tff[i], bs.tff[i - 1], bs.lambda_den);
        }
        r.exact.mul(bs.tff[0], bs.t1);

        if (fflen < tau)
        {
            fflen++;
            r.exact.pow(bs.tff[fflen - 1], bs.lambda_den, delta - 1);
        }

        for (slong j = 0; j < s; j++)
        for (slong k = 0; k < tau; k++)
        for (slong l = 0; l < fflen && l <= k; l++)
            r.exact.madd(bs.Sigma.num IDX(d,j,k), bs.tff[l], bs.M.num IDX(0,j,k-l));
    }

    for (slong n = lo + 1; n < hi; n++)
    {
        assert(bs.z.is_exact);

        // update M
        upoly_evaluate<T>(r.exact, bs.tdeneval, bs.eqden, n);
        for (slong i = 0; i < s; i++)
        for (slong k = 0; k < tau; k++)
            upoly_evaluate<T>(r.exact, bs.tnumeval[i*tau+k], bs.eqnum[i*tau+k], n);

        for (slong j = 0; j < s; j++)
        {
            // update column j
            for (slong k = 0; k < tau; k++)
                r.exact.zero(bs.tff[k]);

            for (slong k = 0; k < tau; k++)
            for (slong h = 0; h <= k; h++)
                r.exact.madd(bs.tff[k], bs.tnumeval[0*tau+h], bs.M.num IDX(0,j,k-h));

            for (slong i = s - 1; i >= 1; i--)
            {
                for (slong k = 0; k < tau; k++)
                for (slong h = 0; h <= k; h++)
                    r.exact.madd(bs.tff[k], bs.tnumeval[i*tau+h], bs.M.num IDX(i,j,k-h));

                for (slong k = 0; k < tau; k++)
                    r.exact.mul(bs.M.num IDX(i,j,k), bs.tdeneval, bs.M.num IDX(i-1,j,k));
            }

            for (slong k = 0; k < tau; k++)
                r.exact.swap(bs.M.num IDX(0,j,k), bs.tff[k]);
        }
        r.exact.mul(bs.M.den, bs.tdeneval);

        // update Σ
        r.exact.mul(bs.t1, bs.tdeneval, bs.z.den);
        r.exact.mul(bs.Sigma.den, bs.t1);
        vec_mul_scalar<T>(r.exact, bs.Sigma.num, bs.t1);
        // Σ.num += zn^(n-lo)*ff*M.num

        // ff = zn^(n-lo)*λd^(δ-1)*(Λ+λ+n-0)*(Λ+λ+n-1)*...*(Λ+λ+n-(d-1))
        fflen = 1;
        r.exact.pow(bs.tff[0], bs.z.num, n - lo);
        r.exact.pow(bs.t1, bs.lambda_den, delta - 1);
        r.exact.mul(bs.tff[0], bs.t1);
        for (slong j = 0; j < s; j++)
        for (slong k = 0; k < tau; k++)
            r.exact.madd(bs.Sigma.num IDX(0,j,k), bs.tff[0], bs.M.num IDX(0,j,k));

        for (slong d = 1; d < delta; d++)
        {
            r.exact.mul(bs.t1, bs.lambda_den, n - (d - 1));
            r.exact.add(bs.t1, bs.lambda_num);
            // ff *= n-(d-1)+λ + Λ
            for (slong i = 0; i < fflen; i++)
                r.exact.divexact(bs.tff[i], bs.lambda_den);

            for (slong i = fflen - 1; i > 0; i--)
            {
                r.exact.mul(bs.tff[i], bs.t1);
                r.exact.madd(bs.tff[i], bs.tff[i - 1], bs.lambda_den);
            }
            r.exact.mul(bs.tff[0], bs.t1);

            if (fflen < tau)
            {
                fflen++;
                r.exact.pow(bs.tff[fflen - 1], bs.z.num, n-lo);
                r.exact.pow(bs.t1, bs.lambda_den, delta - 1);
                r.exact.mul(bs.tff[fflen - 1], bs.t1);
            }

            for (slong j = 0; j < s; j++)
            for (slong k = 0; k < tau; k++)
            for (slong l = 0; l < fflen && l <= k; l++)
                r.exact.madd(bs.Sigma.num IDX(d,j,k), bs.tff[l], bs.M.num IDX(0,j,k-l));
        }
    }

#undef IDX

//std::cout << "bs.Sigma.num: " << bs.Sigma.num << std::endl;
//std::cout << "bs.Sigma.den: " << bs.Sigma.den << std::endl;

//std::cout << "bs.M.num: " << bs.Sigma.num << std::endl;
//std::cout << "bs.M.den: " << bs.Sigma.den << std::endl;

    return;
}


template <class T, class S, bool TA>
void sum_bs_start(
    BiRing<T, S, TA>& R,
    BiRing<T_int_ring_t, S, TA>& ZR,
    SumBsCtx<T_int_ring_t, S, TA>& bs,
    std::vector<upoly<T_elem_t>>& num, slong s, slong tau,
    upoly<T_elem_t>& den,
    BiElem<T, S, TA>& z,
    T_elem_t& lambda,
    slong delta,
    slong N1, slong N0)
{
//std::cout << "sum_bs_start called" << std::endl;
//std::cout << "num: " << num << std::endl;
//std::cout << "den: " << den << std::endl;

    assert(0 <= N0);
    assert(N0 < N1);
    assert(num.size() == s*tau);

    T_elem_t g;

    R.zero(g);
    content<T>(R, g, den);
    for (slong i = 0; i < s*tau; i++)
        content<T>(R, g, num.at(i));

    numerator<T>(ZR, bs.eqden, R, den, g);
    bs.eqnum.resize(s*tau);
    for (slong i = 0; i < s*tau; i++)
        numerator<T>(ZR, bs.eqnum.at(i), R, num.at(i), g);

    R.exact.numerator_denominator(bs.lambda_num, bs.lambda_den, lambda);

//std::cout << "S.eqnum: " << S.eqnum << std::endl;
//std::cout << "S.eqden: " << S.eqden << std::endl;

    if (z.is_exact)
    {
        bs.z.is_exact = true;
        bs.z.have_approx = false;
        R.exact.numerator_denominator(bs.z.num, bs.z.den, z.exact);
    }
    else
    {
        bs.z.is_exact = false;
        bs.z.have_approx = true;
        bs.z.approx = z.approx;
    }

    bs.s = s;
    bs.tau = tau;
    bs.delta = delta;
    bs.Sigma.resize(delta*s*tau);
    bs.M.resize(s*s*tau);
    bs.Delta.resize(delta*s*tau);
    bs.Sigma0.resize(delta*s*tau);
    bs.M0.resize(s*s*tau);
    bs.tmatrix.clear();
    bs.tnumeval.resize(s*tau);
    bs.tff.resize(tau);

    sum_bs_der_recursive(ZR, bs, N1, N0);
}


slong overlap_dist(
    const acb_t a,
    const acb_t Delta,
    const acb_t b,
    slong prec)
{
    if (acb_overlaps(a, b))
        return 0;

    xmag_t m, q;
    acb_get_mag(m.data, a);
    acb_get_mag(q.data, b);
    if (mag_cmp(m.data, q.data) > 0)
        mag_swap(m.data, q.data);
    acb_get_mag(q.data, Delta);
    mag_div(q.data, q.data, m.data);
    if (mag_is_special(q.data))
        return mag_is_zero(q.data) ? 0 : prec;
    fmpz_add_si(&q.data->exp, &q.data->exp, prec);
    if (fmpz_sgn(&q.data->exp) < 0)
        return 0;
    else if (fmpz_cmp_si(&q.data->exp, prec) >= 0)
        return prec;
    else
        return fmpz_get_si(&q.data->exp);
}

template <class S>
slong overlap_dist(
    S& r,
    std::vector<S_elem_t>& a,
    std::vector<S_elem_t>& Delta,
    std::vector<S_elem_t>& b,
    slong length)
{
    slong dist = 0;
    for (slong i = 0; i < length; i++)
        dist = std::max(dist, overlap_dist<S>(r, a.at(i), Delta.at(i), b.at(i)));
    return dist;
}

template <class T, class S, bool TA>
slong sum_bs_continue(
    BiRing<T, S, TA>& r,
    SumBsCtx<T, S, TA>& bs,
    slong hi, slong mid, slong lo)
{
    std::swap(bs.Sigma0, bs.Sigma);
    std::swap(bs.M0, bs.M);
    sum_bs_der_continue<T, S, TA>(r, bs, hi, mid, lo);
    return overlap_dist(bs.Sigma.get_approx(r),
                        bs.Delta.get_approx(r),
                        bs.Sigma0.get_approx(r),
                        bs.delta*bs.s*bs.tau);
}


template <class T, class S, bool TA>
void sum_bs(
    BiRing<T, S, TA>& r,
    S_elem_t& res,
    slong N,
    LEquation<BiElem<T, S, TA>>& E,
    std::vector<BiElem<T, S, TA>>& iv,
    BiElem<T, S, TA>& z)
{
    auto& eq = coeffs_xoutside<BiRing<T, S, TA>>(r, E);
    slong s = bpoly_degree_outside(eq);

    if (N > s)
    {
        T_int_ring_t intre;
        BiRing<T_int_ring_t, S, TA> intr(std::move(intre), r.approx);
        SumBsCtx<T_int_ring_t, S, TA> bs;
        std::vector<upoly<T_elem_t>> numeq(s);
        T_elem_t lambda;

        for (slong i = 0; i < s; i++)
            upoly_neg<BiRing<T, S, TA>>(r, numeq.at(i), eq.coeffs.at(1 + i));

        r.zero(lambda);
        sum_bs_start<T, S, TA>(r, intr, bs, numeq, s, 1, eq.coeffs.at(0),
                               z, lambda, 1, N, s);

        std::vector<S_elem_t>& Sigma = bs.get_sum(intr);

        slong i = 1;
        r.mul(res, Sigma[s - i], iv[i - 1]);
        for (i++; i <= s; i++)
            r.madd(res, Sigma[s - i], iv[i - 1]);
    }
    else
    {
        r.zero(res);
    }

    for (slong i = std::min(N, s); i > 0; i--)
    {
        r.mul(res, z);
        r.add(res, iv.at(i - 1));
    }
}   


/* funny stuff for the evaluation of z^a*log(z)^b */

template <class T, class S, bool TA>
class BranchInfo {
public:
    BiElem<T, S, TA> lambda;
    BiElem<T, S, TA> z;
    BiElem<T, S, TA> invz;
    std::vector<xacb_t> logzpow;
    std::vector<xacb_t> zlogzpow;
    bool have_invz;

    // for -Pi < log(z) <= Pi
    BranchInfo(const BiElem<T, S, TA>& _lambda, const BiElem<T, S, TA>& _z) :
        lambda(_lambda), z(_z), have_invz(false)
    {
        logzpow.resize(1);
        acb_one(logzpow[0].data);
    }

    // for -Pi <= log(z) < Pi
    BranchInfo(const BiElem<T, S, TA>& _lambda, const BiElem<T, S, TA>& _z,
               const BiElem<T, S, TA>& _invz) :
        lambda(_lambda), z(_z), invz(_invz), have_invz(true)
    {
        logzpow.resize(1);
        acb_one(logzpow[0].data);
    }

    void reset(const BiElem<T, S, TA>& _lambda, const BiElem<T, S, TA>& _z)
    {
        lambda = _lambda;
        z = _z;
        have_invz = false;
        logzpow.resize(1);
        acb_one(logzpow[0].data);
    }

    void reset_lambda(const BiElem<T, S, TA>& _lambda)
    {
        lambda = _lambda;
    }

    xacb_t& get_logz(BiRing<T, S, TA>& R);
    void zpow(BiRing<T, S, TA>& R, xacb_t& res, slong i);
    void zpowlogpow(BiRing<T, S, TA>& R, xacb_t& res, slong i, ulong j);
    xacb_t& get_logzpow(BiRing<T, S, TA>& R, slong j);
    xacb_t& get_zlogzpow(BiRing<T, S, TA>& R, slong j);
    void zpowlogpoly_eval(BiRing<T, S, TA>& R, xacb_t& res, slong n, std::vector<S_elem_t>& e);
};

template<class T, class S, bool TA>
xacb_t& BranchInfo<T, S, TA>::get_logz(BiRing<T, S, TA>& R)
{
    slong prec = R.approx.precision();

    if (logzpow.size() > 1)
        return logzpow[1];

    logzpow.resize(2);
    xacb_t& l = logzpow[1];
    if (have_invz)
    {
        R.approx.log(l, R.approx_value(invz));
        R.approx.neg(l);
    }
    else
    {
        R.approx.log(l, R.approx_value(z));
    }

    return l;
}

/* z^(λ+i) */
template<class T, class S, bool TA>
void BranchInfo<T, S, TA>::zpow(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    slong i)
{
    if (R.is_zero(lambda))
    {
        if (have_invz)
            R.approx.pow(res, R.approx_value(invz), -i);
        else
            R.approx.pow(res, R.approx_value(z), i);
        return;
    }

    S_elem_t t;
    R.approx.add(t, R.approx_value(lambda), i);
    if (have_invz)
    {
        R.approx.neg(t);
        R.approx.pow(res, R.approx_value(invz), t);
    }
    else
    {
        R.approx.pow(res, R.approx_value(z), t);
    }
}

// Given |z| <= zmag and |arg z| <= zarg, return a ball for z^A*log(z)^j/j!
void fancy_pow_at_zero(
    acb_t res,
    arb_t zmag, arb_t zarg,
    acb_t a,
    ulong j,
    slong prec)
{
    assert(arb_is_nonnegative(zmag));
    assert(arb_is_nonnegative(zarg));
    xarb_t l, t, r, aoj, joa;
    if (arb_is_positive(acb_realref(a)))
    {
        if (arb_is_zero(zmag))
        {
            arb_zero(r.data);
        }
        else if (j == 0)
        {
            arb_pow(r.data, zmag, acb_realref(a), prec);
        }
        else
        {
            arb_div_ui(aoj.data, acb_realref(a), j, prec);
            arb_log(l.data, zmag, prec);
            arb_pow(t.data, zmag, aoj.data, prec);
            arb_mul(r.data, t.data, l.data, prec);
            arb_abs(r.data, r.data);
            // if !(-l > j/a)
            arb_ui_div(joa.data, j, acb_realref(a), prec);
            arb_neg(l.data, l.data);
            if (!arb_gt(l.data, joa.data))
            {
                arb_mul_2exp_si(joa.data, joa.data, -1);
                arb_max(r.data, r.data, joa.data, prec);
            }
            arb_addmul(r.data, t.data, zarg, prec);
            arb_pow_ui(r.data, r.data, j, prec);
            arb_fac_ui(t.data, j, prec);
            arb_div(r.data, r.data, t.data, prec);
        }
        arb_mul(t.data, acb_imagref(a), zarg, prec);
        arb_abs(t.data, t.data);
        arb_exp(t.data, t.data, prec);
        arb_mul(r.data, r.data, t.data, prec);
        arb_set(acb_realref(res), r.data);
        arb_swap(acb_imagref(res), r.data);
    }
    else if (j == 0 && acb_is_zero(a))
    {
        acb_one(res);
    }
    else
    {
        acb_indeterminate(res);
    }
}

/* z^(λ+i)*log(z)^j/j! */
template<class T, class S, bool TA>
void BranchInfo<T, S, TA>::zpowlogpow(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    slong i, ulong j)
{
    zpow(R, res, i);
    if (j < 1)
        return;

    xacb_t t1, t2;
    xarb_t t3, re, im;
    xacb_t& l = get_logz(R);

    slong prec = R.approx.precision();

    if (acb_is_finite(l.data))
    {
        acb_pow_ui(t2.data, l.data, j, prec);
        acb_mul(t1.data, res.data, t2.data, prec);
        arb_fac_ui(t3.data, j, prec);
        acb_div_arb(res.data, t1.data, t3.data, prec);
    }
    else
    {
        R.approx.add(t1, R.approx_value(lambda), i);
        R.approx.abs(re, R.approx_value(z));
        arb_const_pi(im.data, 60);
        fancy_pow_at_zero(res.data, re.data, im.data, t1.data, j, prec);
    }

//std::cout << "res: " << res << std::endl;
}

// log(z)^j/j!
template <class T, class S, bool TA>
xacb_t& BranchInfo<T, S, TA>::get_logzpow(
    BiRing<T, S, TA>& R,
    slong j)
{
    assert(j >= 0);
    if (j < 1)
        return logzpow.at(0);

    get_logz(R);

    slong prec = R.approx.precision();

    while (j >= logzpow.size())
    {
        ulong n = logzpow.size();
        logzpow.emplace_back();
        acb_pow_ui(logzpow.back().data, logzpow[1].data, n, prec);
        xarb_t f;
        arb_fac_ui(f.data, n, prec);
        acb_div_arb(logzpow.back().data, logzpow.back().data, f.data, prec);
    }

    return logzpow.at(j);
}

// z*log(z)^j/j!
template <class T, class S, bool TA>
xacb_t& BranchInfo<T, S, TA>::get_zlogzpow(
    BiRing<T, S, TA>& R,
    slong j)
{
    assert(j >= 0);

    // z = get_approx(Z, Φ)
    // j<1 && return z
    xacb_t& z1 = R.approx_value(z);
    if (j < 1)
        return z1;

    slong prec = R.approx.precision();

    xacb_t& l = get_logz(R);
    if (acb_is_finite(l.data))
    {
        while (j > zlogzpow.size())
        {
            zlogzpow.emplace_back();
            ulong n = logzpow.size();
            acb_pow_ui(zlogzpow.back().data, l.data, n, prec);
            acb_mul(zlogzpow.back().data, zlogzpow.back().data, z1.data, prec);
            xarb_t f;
            arb_fac_ui(f.data, n, prec);
            acb_div_arb(zlogzpow.back().data, zlogzpow.back().data, f.data, prec);
        }
    }
    else
    {
        xarb_t r, pi;
        xacb_t one;
        acb_abs(r.data, z1.data, prec);
        arb_const_pi(pi.data, 60);
        acb_one(one.data);
        while (j > zlogzpow.size())
        {
            zlogzpow.emplace_back();
            fancy_pow_at_zero(zlogzpow.back().data, r.data, pi.data, one.data,
                              zlogzpow.size(), prec);
        }
    }

    return zlogzpow.at(j - 1);
}

// z^(λ+n)*Σ_{j<τ}e_j*Λ^(τ-1-j) with Λ^(τ-1-j) replaced by log(z)^j/j!
template <class T, class S, bool TA>
void BranchInfo<T, S, TA>::zpowlogpoly_eval(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    slong n,
    std::vector<S_elem_t>& e)
{
    slong tau = e.size();
    slong j = tau - 1;
    if (j < 1)
    {
        zpow(R, res, n);
        R.approx.mul(res, e[0]);
        return;
    }

    bool ok = acb_is_finite(get_logz(R).data);
    xacb_t& p = ok ? get_logzpow(R, j) : get_zlogzpow(R, j);
    R.approx.mul(res, p, e[tau - 1 - j]);
    while (--j > 0)
        R.approx.madd(res, p, e.at(tau - 1 - j));

    if (ok)
        R.approx.add(res, e.at(tau - 1));
    else
        R.approx.madd(res, e.at(tau - 1), R.approx_value(z));

    zpow(R, R.approx.tmp_xacb[0], n - !ok);
    R.approx.mul(res, R.approx.tmp_xacb[0]);
}


/* evaluation of series solutions to diff equations */


template <class T>
void next_u_coeffs(
    T& R,
    std::vector<std::vector<std::vector<T_elem_t>>>& u,
    slong & tau,
    LEquation<T_elem_t>& P,
    T_elem_t& lambda,
    std::vector<xfmpz_t>& ns,
    slong & noff,
    slong N)
{
//std::cout << "---- next_u_coeffs called ------------" << std::endl;
//std::cout << "tau: " << tau << std::endl;
//std::cout << "u: " << u << std::endl;
//std::cout << "P: " << P << std::endl;

    slong nu = ns.size();
    T_elem_t lambdaN;
    R.add(lambdaN, lambda, N);
    bpoly<T_elem_t> Pn;
    bpoly_taylor_shift_inside<T>(R, Pn, coeffs_xoutside<T>(R, P), lambdaN);

//std::cout << "Pn: " << Pn << std::endl;

    slong s = bpoly_degree_outside(Pn);

//std::cout << "s: " << s << std::endl;

//std::cout << "iv: " << std::endl;

    std::vector<std::vector<T_elem_t>> uN;
    slong mu = 0;
    while (noff < nu && fmpz_equal_si(ns[noff].data, N))
    {
        uN.push_back(std::vector<T_elem_t>(nu));
        for (slong l = 0; l < nu; l++)
            R.set(uN.at(mu).at(l), slong(l == noff));
        mu++;
        noff++;
    }

//std::cout << "uN: " << uN << std::endl;

//std::cout << "mu: " << mu << std::endl;

    // mu = number of initial conditions at lambda + N = multiplicity of root
    for (slong i = 0; i < mu; i++)
        assert(i >= Pn.coeffs.at(0).coeffs.size() || R.is_zero(Pn.coeffs.at(0).coeffs.at(i)));

//std::cout << "rhs" << std::endl;
    uN.resize(mu + tau);
    for (slong j = 0; j < tau; j++)
    {
        uN.at(mu + j).resize(nu);
        for (slong l = 0; l < nu; l++)
            R.zero(uN.at(mu + j).at(l));
    }

    for (slong i = 0; i < s; i++)
    {
        auto & Pni = Pn.coeffs.at(i + 1).coeffs;
        for (slong k = 0; k < Pni.size(); k++)
        for (slong j = 0; j + k < u.at(i).size(); j++)
        for (slong l = 0; l < nu; l++)
            R.msub(uN.at(mu + j).at(l),
                          u.at(i).at(j + k).at(l),
                          Pni.at(k));
    }

//std::cout << "uN: " << uN << std::endl;

//std::cout << "lhs" << std::endl;
    for (slong i = 1; i <= tau; i++)
    {
        for (slong j = 1; j < i; j++)
        for (slong l = 0; l < nu; l++)
            R.msub(uN.at(mu + tau - i).at(l),
                          uN.at(mu + tau - i + j).at(l),
                          Pn.coeffs.at(0).coeffs.at(mu + j));

        for (slong l = 0; l < nu; l++)
            R.divexact(uN.at(mu + tau - i).at(l),
                       uN.at(mu + tau - i).at(l),
                       Pn.coeffs.at(0).coeffs.at(mu));
    }

//std::cout << "triming zeros" << std::endl;
    slong k = mu;
    for (; k > 0; k--)
    {
        bool allzero = true;
        for (slong l = 0; l < nu; l++)
            allzero = allzero && R.is_zero(uN.at(tau + k - 1).at(l));
        if (!allzero)
            break;
    }
    tau += k;
    uN.resize(tau);

    std::swap(u.at(s - 1), uN);
    for (slong i = s - 1; i > 0; i--)
        std::swap(u.at(i), u.at(i - 1));

//std::cout << "next_u_coeffs returning" << std::endl;
//std::cout << "tau: " << tau << std::endl;
//std::cout << "u: " << u << std::endl;
}

template <class T>
void u_derivative(
    T& R,
    std::vector<std::vector<T_elem_t>>& uN,
    T_elem_t& fac,
    slong tau,
    slong nu)
{
    for (slong j = 0; j < tau; j++)
    for (slong l = 0; l < nu; l++)
    {
        R.mul(uN[j][l], uN[j][l], fac);
        if (j + 1 < tau)
            R.add(uN[j][l], uN[j][l], uN[j + 1][l]);
    }
}


template <class T, class S, bool TA>
void eval_basis_forward(
    BiRing<T, S, TA>& R,
    xacb_mat_t& M,              /* result */
    LEquation<BiElem<T, S, TA>>& P,
    BranchInfo<T, S, TA>& Z,
    std::vector<xfmpz_t>& ns,
    std::vector<BiElem<T, S, TA>>& alphas)
{
    slong prec = R.approx.precision();
    slong delta = M.nrows();
    slong nu = ns.size();   // number of initial conditions
    slong tau = 0;          // strict bound on the power of log(z) thus far
    slong maxN = 20*prec;   // max number of terms to sum
    assert(fmpz_is_zero(ns[0].data));
    assert(M.ncols() == ns.size());

    auto& Ptheta = coeffs_xoutside<BiRing<T, S, TA>>(R, P);

//std::cout << "P: " << P << std::endl;

    slong s = bpoly_degree_outside<BiElem<T, S, TA>>(Ptheta);

//std::cout << "s: " << s << std::endl;

    // u is a nested array of the last s solutions
    assert(s > 0 && nu > 0);
    std::vector<std::vector<std::vector<BiElem<T, S, TA>>>> u(s);
    std::vector<std::vector<BiElem<T, S, TA>>> uN;
    std::vector<BiElem<T, S, TA>> uN0;
    BiElem<T, S, TA> lambdaN;
    xacb_t t1, t2, ts;

    slong N = 0;

    // M is the matrix answer
    acb_mat_zero(M.data);

    // the nsoff reads through the ns as we sum past them
    slong nsoff = 0;

    int changed_count = 0;
    while (true)
    {
        bool may_stop = nsoff >= nu;
        assert(N <= maxN);

//std::cout << "N: " << N << std::endl;

        next_u_coeffs<BiRing<T, S, TA>>(R, u, tau, P, Z.lambda, ns, nsoff, N);
        uN = u[0];

//std::cout << "uN: " << uN << std::endl;

        // add uN*z^N to sum
        bool changed = false;
        for (slong d = 0; d < delta; d++)
        {
            if (d > 0)
            {
                R.add(lambdaN, Z.lambda, N - d + 1);
                u_derivative<BiRing<T, S, TA>>(R, uN, lambdaN, tau, nu);
            }

            for (slong l = 0; l < nu; l++)
            {
                R.approx.zero(ts);
                for (slong j = 0; j < tau; j++)
                {
                    if (R.is_zero(uN[j][l]))
                        continue;
                    Z.zpowlogpow(R, t2, N - d, j);
                    R.approx.madd(ts, t2, R.approx_value(uN[j][l]));
                }

                R.approx.add(ts, &M[d][l]);
                changed = changed || !acb_overlaps(ts.data, &M[d][l]);
                acb_swap(&M[d][l], ts.data);
            }
        }

        N++;

        if (!may_stop)
            continue;

        if (!changed)
        {
            if (++changed_count > 10)
                break;
        }
        else
        {
            changed_count = 0;
        }
    }

    std::vector<xarb_poly_t> Er;
#if 0
    tail_bound<T>(R, Er, P, u, ns, nsoff, alphas, Z, delta, N, tau, nu, prec);
#endif
    xfmpz_t fd(1);
    for (slong d = 0; d < delta; d++)
    {
        if (d > 1)
            fmpz_mul_si(fd.data, fd.data, d);

        for (slong l = 0; l < nu; l++)
        {
            if (d > 1)
                acb_div_fmpz(&M[d][l], &M[d][l], fd.data, prec);
#if 0
            if (d < Er[l].data->length)
            {
                arb_add_error(acb_realref(&M[d][l]), Er[l].data->coeffs + d);
                arb_add_error(acb_imagref(&M[d][l]), Er[l].data->coeffs + d);
            }
#endif
        }
    }
}

template <class T, class S, bool TA>
void compute_pfq_near_zero_taylor(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    std::vector<BiElem<T, S, TA>> a,
    std::vector<BiElem<T, S, TA>> b,
    BiElem<T, S, TA> z)
{
    size_t p = a.size();
    size_t q = b.size();
    assert(q + 1 >= p);

    LEquation<BiElem<T, S, TA>> eq;

//std::cout << "a: " << a << std::endl;
//std::cout << "b: " << b << std::endl;

    pfq_equ_at_zero<BiRing<T, S, TA>>(R, eq, a, b);

//std::cout << "eq: " << eq << std::endl;


    std::vector<BiElem<T, S, TA>> roots(q + 1);
    R.zero(roots[0]);
    for (size_t i = 0; i < q; i++)
        R.sub(roots[i + 1], 1, b[i]);

    eval_basis<T, S, TA>(R, e, eq, Z, offs, roots);

template <class T, class S, bool TA>
void compute_pfq_anywhere_taylor(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    std::vector<BiElem<T, S, TA>> a,
    std::vector<BiElem<T, S, TA>> b,
    BiElem<T, S, TA> z)
{
    size_t p = a.size();
    size_t q = b.size();
    assert(q + 1 == p);
    slong k = 0;

    LEquation<BiElem<T, S, TA>> eq;
    pfq_equ_at_zero_quadratic<BiRing<T, S, TA>>(R, eq, k, a, b);

    S_elem_t w, s, t, t1;
    BiElem<T, S, TA> u;
    R.sub(u, 1, z);
    R.approx.sqrt(t, R.approx_value(u));
    R.approx.add(s, t, 1);
    R.approx.sub(t1, 1, t);
    R.approx.divexact(w, t1, s);

    std::vector<BiElem<T, S, TA>> roots(q + 1);
    R.zero(roots[0]);
    for (size_t i = 0; i < q; i++)
        R.sub(roots[i + 1], 1, b[i]);
    std::vector<xfmpz_t> offs(1);
    fmpz_zero(offs[0].data);


    BranchInfo<T, S, TA> Z(roots[0], w);
    xacb_mat_t e(1, 1);
    eval_basis<T>(R, e, eq, Z, offs, roots);

    R.approx.mul(t, R.approx_value(a[k]), -2);
    R.approx.mul_2exp(s, -1);
    R.approx.pow(t1, s, t);
    R.approx.mul(res, &e[0][0], t1);
}

/*
for each s = λ+ns[j] compute the coeff of z^s in
residue Γ(B)/Γ(A)*Γ(A+s)*Γ(-s)/Γ(B+s)*z^-s,  A=(λ+ns,arest), B=(b)
the coeff is returned as the coeffs of 1, log(z), ..., log^(μ-1)(z)/(μ-1)
TODO: currently doesn't work in terminating case
*/
template <class T, class S, bool TA>
void pfq_initial_values_at_infinity(
    BiRing<T, S, TA>& R,
    std::vector<S_elem_t>& iv,
    BiElem<T, S, TA>& lambda,
    std::vector<xfmpz_t>& ns,
    std::vector<BiElem<T, S, TA>*> arest,
    std::vector<BiElem<T, S, TA>> a,
    std::vector<BiElem<T, S, TA>> b)
{
    slong m = ns.size();
    BiElem<T, S, TA> t, lambdan;
    S_elem_t t0, t1;
    S_poly_elem_t s, s0, s1, s2;
    xfmpz_t d, ZZlambda, intd, td;
    bs_product<S, S_elem_t> f(R.approx, t0);
    iv.resize(m);

    for (auto& bi : b)
    {
        R.approx.gamma(t0, R.approx_value(bi));
        f.push(R.approx, t0);
    }
    for (auto ai : arest)
    {
        R.approx.rgamma(t0, R.approx_value(*ai));
        f.push(R.approx, t0);
    }

    if (m < 2)
    {
        assert(m == 1);
        for (auto& bi : b)
        {
            R.sub(lambdan, bi, lambda);
            R.approx.rgamma(t0, R.approx_value(lambdan));
            f.push(R.approx, t0);
        }
        for (auto ai : arest)
        {
            R.sub(lambdan, *ai, lambda);
            R.approx.gamma(t0, R.approx_value(lambdan));
            f.push(R.approx, t0);
        }

        if (f.is_empty())
            R.approx.one(iv[0]);
        else
            R.approx.swap(iv[0], f.product());

        return;
    }

    bool int_case = R.is_integer(ZZlambda, lambda);

    for (auto& n : ns)
    {
        R.add(lambdan, lambda, n);
        R.approx.rgamma(t0, R.approx_value(lambdan));
        f.push(R.approx, t0);
    }

    for (slong mu, j = 0; j < m; j += mu)
    {
        const xfmpz_t& n = ns[j];
        // μ = multiplicity of exponent λ+n
        mu = 1;
        while (j + mu < m && fmpz_equal(ns[j + mu].data, n.data))
            mu++;

        // compute coeffs iv[j], ..., iv[j+μ-1] of z^(λ+n), ..., z^(λ+n)*log^(μ-1)(z)/(μ-1)!
        // there is a pole of order j + mu at s = -λ-n + ε
        slong ord = j + mu;
        slong sign = 0;

        R.approx.one(t1);
        R.approx.gamma_series(s, t1, 1, ord);
        R.approx.pow_series(s0, s, m - int_case, ord);

        // handle Γ(-s) = Γ(λ+n - ε)
        R.add(lambdan, lambda, n);
        fmpz_zero(intd.data);
        if (int_case)
        {
            // Γ(d + 1 - ε) = π*ε*csc(π*ε)*(1 - ε)_d / Γ(1 + ε)
            fmpz_add(intd.data, ZZlambda.data, n.data);
            fmpz_sub_ui(intd.data, intd.data, 1);
            R.approx.cscpi_series(s1, ord);
        }
        else
        {
            R.approx.gamma_series(s1, R.approx_value(lambdan), -1, ord);
        }
        R.approx.mul_series(s, s0, s1, ord);

        // handle non-poles from s = -λ-n + ε
        R.approx.one(s1);
        for (slong k = j + mu; k < m; k++)
        {
            fmpz_sub(d.data, ns[k].data, n.data);
            fmpz_sub_ui(d.data, d.data, 1);
            assert(fmpz_sgn(d.data) >= 0);
            if (fmpz_sgn(d.data) <= 0)
                continue;
            R.approx.rising_factorial_series(s0, t1, 1, d, ord);
            R.approx.mul_series(s2, s1, s0, ord);
            R.approx.swap(s1, s2);
        }

        // handle poles from s = -λ-n + ε
        for (slong k = 0; k < j + mu; k++)
        {
            fmpz_sub(d.data, n.data, ns[k].data);
            assert(fmpz_sgn(d.data) >= 0);
            sign += !fmpz_is_even(d.data);
            if (fmpz_sgn(d.data) <= 0)
                continue;
            fmpz_sub(td.data, intd.data, d.data);
            int tdsgn = fmpz_sgn(td.data);
            if (tdsgn > 0)
            {
                fmpz_add_ui(d.data, d.data, 1);
                R.approx.set(t0, d);
                R.approx.rising_factorial_series(s0, t0, -1, td, ord);
                R.approx.mul_series(s2, s1, s0, ord);
                R.approx.swap(s1, s2);
            }
            else if (tdsgn < 0)
            {
                fmpz_add_ui(d.data, intd.data, 1);
                fmpz_neg(td.data, td.data);
                R.approx.set(t0, d);
                R.approx.rising_factorial_series(s0, t0, -1, td, ord);
                R.approx.div_series(s2, s1, s0, ord);                
                R.approx.swap(s1, s2);
            }
            fmpz_zero(intd.data);
        }

        if (fmpz_sgn(intd.data) > 0)
        {
            R.approx.rising_factorial_series(s0, t1, -1, intd, ord);
            R.approx.mul_series(s2, s1, s0, ord);
            R.approx.swap(s1, s2);
        }

        R.approx.mul_series(s0, s, s1, ord);
        R.approx.swap(s, s0);

        // handle non-poles of Γ(arest[k]+s) = Γ(arest[k]-λ-n + ε)
        for (auto ai : arest)
        {
            R.sub(t, *ai, lambdan);
            R.approx.gamma_series(s0, R.approx_value(t), 1, ord);
            R.approx.mul_series(s1, s, s0, ord);
            R.approx.swap(s1, s);
        }

        // handle Γ(b[k]+s) = Γ(b[k]-λ-n + ε)
        for (auto& bi : b)
        {
            R.sub(t, bi, lambdan);
            R.approx.rgamma_series(s0, R.approx_value(t), 1, ord);
            R.approx.mul_series(s1, s, s0, ord);
            R.approx.swap(s1, s);
        }

        // now we have s = ε^-ord*(s[0] + ... s[ord-1]*ε^(ord-1) + O(s^ord))
        //     determine r[j+k] for 0 <= k < μ.
        // Sum[s[0]*ε^j,{j,0,inf}] * Sum[(-1)^k*Log[z]^k/k!*ε^k,{k,0,inf}]
        for (slong k = 0; k < mu; k++)
        {
            // coeff Log[z]^k/k!*ε^(ord-1) has j = ord-1-k
            assert(k < ord);
            if (ord - 1 - k >= s.length())
            {
                R.approx.zero(iv[j + k]);
                continue;
            }

            if (f.is_empty())
                R.approx.set(iv[j + k], s.coeff(ord - 1 - k));
            else
                R.approx.mul(iv[j + k], f.product(), s.coeff(ord - 1 - k));

            if ((sign + k)&1)
                R.approx.neg(iv[j + k]);
        }
    }
}

template <class E>
using tri_tuple = std::tuple<E, std::vector<xfmpz_t>, std::vector<slong>>;


template <class T>
std::vector<tri_tuple<T_elem_t>> partition_mod1(
    T& R,
    std::vector<T_elem_t>& a)
{
    std::vector<tri_tuple<T_elem_t>> r;
    T_elem_t t;
    xfmpz_t k;

    for (slong i = 0; i < a.size(); i++)
    {
        for (slong j = 0; j < r.size(); j++)
        {
            R.sub(t, a[i], std::get<0>(r[j]));
            if (R.is_integer(k, t))
            {
                std::get<1>(r[j]).push_back(std::move(k));
                std::get<2>(r[j]).push_back(i);
                goto continue_outside;
            }
        }
        r.emplace_back(a[i], std::vector<xfmpz_t>{xfmpz_t(0)}, std::vector<slong>{i});
    continue_outside:;
    }

    for (auto & rj : r)
    {
        auto & rj1 = std::get<1>(rj);
        std::sort(rj1.begin(), rj1.end());
        R.add(std::get<0>(rj), std::get<1>(rj)[0].data);
        for (slong l = rj1.size(); l > 0 ; l--)
            fmpz_sub(rj1[l - 1].data, rj1[l - 1].data, rj1[0].data);
    }
    return r;
}

template <class T, class S, bool TA>
void compute_pfq_near_infinity_taylor(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    std::vector<BiElem<T, S, TA>>& a,
    std::vector<BiElem<T, S, TA>>& b,
    BiElem<T, S, TA>& z)
{
//std::cout << "compute_pfq_near_infinity_taylor called" << std::endl;
    slong p = a.size();
    slong q = b.size();
    assert(q + 1 <= p);

    std::vector<BiElem<T, S, TA>*> arest(p);
    std::vector<S_elem_t> iv(p);
    xacb_mat_t e(1, 1);

    BiElem<T, S, TA> nz, rz;
    R.neg(nz, z);
    R.inv(rz, nz);
    BranchInfo<T, S, TA> Z(a[0], rz, nz); // dummy lambda

    LEquation<BiElem<T, S, TA>> eq;
    pfq_equ_at_infinity<BiRing<T, S, TA>>(R, eq, a, b);

    R.approx.zero(res);
    std::vector<tri_tuple<BiElem<T, S, TA>>> I(partition_mod1<BiRing<T, S, TA>>(R, a));
    for (auto& i : I)
    {
        BiElem<T, S, TA>& lambda = std::get<0>(i);
        std::vector<xfmpz_t>& ns = std::get<1>(i);
        std::vector<slong>& indices = std::get<2>(i);
        arest.clear();
        for (slong j = 0; j < p; j++)
            if (std::find(indices.begin(), indices.end(), j) == indices.end())
                arest.push_back(&a[j]);
        pfq_initial_values_at_infinity<T, S, TA>(R, iv, lambda, ns, arest, a, b);
        e.resize(1, ns.size());
        Z.reset_lambda(lambda);
        eval_basis<T, S, TA>(R, e, eq, Z, ns, a);
        for (slong j = 0; j < ns.size(); j++)
            R.approx.madd(res, &e[0][j], iv[j]);
    }
}


template <class T, class S, bool TA>
void compute_pfq_near_one_taylor(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    std::vector<BiElem<T, S, TA>>& a,
    std::vector<BiElem<T, S, TA>>& b,
    BiElem<T, S, TA>& z)
{
    slong n = a.size();
    assert(n == b.size() + 1);

    BiElem<T, S, TA> w, t;
    R.one(t);
    R.divexact(w, t, std::min(n, slong(8)));

    // 0 -> 1-w
    LEquation<BiElem<T, S, TA>> eq;
    pfq_equ_at_zero<BiRing<T, S, TA>>(R, eq, a, b);

    std::vector<BiElem<T, S, TA>> roots(n);
    R.zero(roots[0]);
    for (slong i = 1; i < n; i++)
        R.sub(roots[i], 1, b[i - 1]);

    std::vector<xfmpz_t> offs(1);
    fmpz_zero(offs[0].data);
    R.sub(t, 1, w);
    BranchInfo<T, S, TA> Z(roots[0], t);

    xacb_mat_t e0(n, 1);
    xacb_mat_t e1(1, n);
    xacb_mat_t e2(n, n);
    xacb_mat_t et(n, 1);
    xacb_mat_t ei(n, 1);

    eval_basis<T, S, TA>(R, e0, eq, Z, offs, roots);
    for (slong i = 1; i < n; i += 2)
        R.approx.neg(&e0[i][0]);

    BiElem<T, S, TA> onemz;
    R.sub(onemz, 1, z);

    pfq_equ_at_one<BiRing<T, S, TA>>(R, eq, a, b);

    // roots of indicial equation are sigma and 0:n-2
    R.neg(roots[0], a[n - 1]);
    for (slong i = 0; i < n - 1; i++)
    {
        R.set(roots.at(i + 1), i);
        R.add(roots[0], b[i]);
        R.sub(roots[0], a[i]);
    }

    offs.resize(n);
    for (slong i = 0; i < n - 1; i++)
        fmpz_set_ui(offs[i].data, i);

    bool intcase = R.is_integer(offs[n - 1], roots[0]);
    if (n < 2 || intcase)
    {
        std::sort(offs.begin(), offs.end());

        if (n < 2)
            R.set(t, roots[0]);
        else
            R.set(t, offs[0]);

        for (slong i = n - 1; i >= 0; i--)
            fmpz_sub(offs[i].data, offs[i].data, offs[0].data);

        Z.reset(t, w);
        eval_basis<T, S, TA>(R, e2, eq, Z, offs, roots);

        Z.reset(t, onemz);
        eval_basis<T, S, TA>(R, e1, eq, Z, offs, roots);

        et.resize(1, 1);
    }
    else
    {
        e2.hshrink(n - 1);
        e1.hshrink(n - 1);
        offs.resize(n - 1);
        Z.reset(roots.at(1), w);
        eval_basis<T, S, TA>(R, e2, eq, Z, offs, roots);

        Z.reset(roots.at(1), onemz);
        eval_basis<T, S, TA>(R, e1, eq, Z, offs, roots);

        offs.resize(1);
        Z.reset(roots.at(0), w);
        eval_basis<T, S, TA>(R, et, eq, Z, offs, roots);
        e2.hcat(et.data);

        et.resize(1, 1);
        Z.reset(roots.at(0), onemz);
        eval_basis<T, S, TA>(R, et, eq, Z, offs, roots);
        e1.hcat(et.data);
    }

    // e1 * e2^-1 * e0
    slong prec = R.approx.precision();
    if (0 == acb_mat_solve(ei.data, e2.data, e0.data, prec))
    {
        acb_indeterminate(res.data);
        return;
    }
    acb_mat_mul(et.data, e1.data, ei.data, prec);
    acb_swap(res.data, &et[0][0]);
}


template <class T, class S, bool TA>
void compute_pfq_plain_terms_horner(
    BiRing<T, S, TA>& R,
    S_elem_t& res,
    std::vector<BiElem<T, S, TA>>& a,
    std::vector<BiElem<T, S, TA>>& b,
    BiElem<T, S, TA>& z,
    slong N)
{
    BiElem<T, S, TA> t1, t2, t3;
    upoly<BiElem<T, S, TA>> A, B;

    upoly_set<BiRing<T, S, TA>>(R, A, z);
    for (auto & ai : a)
        upoly_mul_linear<BiRing<T, S, TA>>(R, A, ai);

    upoly_one<BiRing<T, S, TA>>(R, B);
    for (auto & bi : b)
        upoly_mul_linear<BiRing<T, S, TA>>(R, B, bi);

    R.approx.one(res);
    for (slong i = N - 1; i > 0; i--)
    {
        upoly_evaluate<BiRing<T, S, TA>>(R, t1, A, i - 1);
        upoly_evaluate<BiRing<T, S, TA>>(R, t2, B, i - 1);
        R.mul(t2, i);
        R.divexact(t3, t1, t2);
        R.mul(res, t3);
        R.approx.add(res, 1);
    }
}

template <class T>
slong pfq_terminates_test(
    T& R,
    std::vector<T_elem_t>& a,
    slong limit)
{
    ulong N = -UWORD(1);
    xfmpz_t i;
    for (auto & ai : a)
    {
        if (R.is_integer(i, ai) &&
            fmpz_sgn(i.data) <= 0 &&
            fmpz_cmp_si(i.data, -limit) >= 0)
        {
            N = std::min(N, ulong(1) - fmpz_get_si(i.data));
        }
    }
    return slong(N);
}

template <class T, class S, bool TA>
void compute_pfq(
    BiRing<T, S, TA>& R,
    xacb_t& res,
    std::vector<BiElem<T, S, TA>>& a,
    std::vector<BiElem<T, S, TA>>& b,
    BiElem<T, S, TA>& z)
{
    auto zz = R.approx.template convert<std::complex<double>>(R.approx_value(z));
    double x = FLINT_MAX(FLINT_MIN(zz.real(), 1e5), -1e5);
    double y = FLINT_MAX(FLINT_MIN(zz.imag(), 1e5), -1e5);

//std::cout << "compute_pfq called x = " << x << ", y = " << y << std::endl;

    slong N = pfq_terminates_test<BiRing<T, S, TA>>(R, a, R.approx.precision());
    if (N > 0)
    {
        compute_pfq_plain_terms<T, S, TA>(R, res, a, b, z, N);
        return;
    }

    slong p = a.size();
    slong q = b.size();
    slong d = q + 1 - p;
    if (d > 0)
    {
        compute_pfq_near_zero_taylor<T, S, TA>(R, res, a, b, z);
        return;
    }
    else if (d < 0)
    {
        compute_pfq_near_infinity_taylor<T, S, TA>(R, res, a, b, z);
        return;
    }
    else
    {
        if (x*x + y*y < 0.7)
            compute_pfq_near_zero_taylor<T, S, TA>(R, res, a, b, z);
        else if (x*x + y*y > 1.3)
            compute_pfq_near_infinity_taylor<T, S, TA>(R, res, a, b, z);
        else if (x*x + y*y < 2*x - 0.5) // 2*x + r^2-1
            compute_pfq_near_one_taylor<T, S, TA>(R, res, a, b, z);
        else
            compute_pfq_anywhere_taylor<T, S, TA>(R, res, a, b, z);
        return;
    }
}


bool _try_fmpq(xacb_t& res, er e, slong prec)
{
    std::vector<xfmpq_t> a, b;
    xfmpq_t z;

    if (!from_ex(z, echild(e,3)) || !from_ex(a, echild(e,1)) || !from_ex(b, echild(e,2)))
        return false;

    rfmpq_t r1;
    rarb_t r2(prec + 60);
    BiRing<rfmpq_t, rarb_t, false> R(r1, r2);

    BiElem<rfmpq_t, rarb_t, false> Z;
    std::vector<BiElem<rfmpq_t, rarb_t, false>> A(a.size()), B(b.size());

    R.set(Z, z);
    for (size_t i = 0; i < a.size(); i++)
        R.set(A[i], a[i]);
    for (size_t i = 0; i < b.size(); i++)
        R.set(B[i], b[i]);

    compute_pfq<rfmpq_t, rarb_t, false>(R, res, A, B, Z);
    return true;
}

bool _try_fmpqi(xacb_t& res, er e, slong prec)
{
    std::vector<xfmpqi_t> a, b;
    xfmpqi_t z;

    if (!from_ex(z, echild(e,3)) || !from_ex(a, echild(e,1)) || !from_ex(b, echild(e,2)))
        return false;

    rfmpqi_t r1;
    racb_t r2(prec + 60);
    BiRing<rfmpqi_t, racb_t, false> R(r1, r2);

    BiElem<rfmpqi_t, racb_t, false> Z;
    std::vector<BiElem<rfmpqi_t, racb_t, false>> A(a.size()), B(b.size());

    R.set(Z, z);
    for (size_t i = 0; i < a.size(); i++)
        R.set(A[i], a[i]);
    for (size_t i = 0; i < b.size(); i++)
        R.set(B[i], b[i]);

    compute_pfq<rfmpqi_t, racb_t, false>(R, res, A, B, Z);
    return true;
}

ex ncode_sHypergeometricPFQ(er e, slong prec)
{
//std::cout << "***********************************************" << std::endl;
//std::cout << "ncode_sHypergeometricPFQ(" << prec << "): " << e << std::endl;
    if (!ehas_head_sym_length(e, gs.sym_sHypergeometricPFQ.get(), 3))
        return ecopy(e);

    xacb_t res;
    if (!_try_fmpq(res, e, prec) &&
        !_try_fmpqi(res, e, prec))
    {
        return ecopy(e);
    }

//std::cout << "res: " << res << std::endl;
    return emake_cmplx_move(res.data);
}


ex ncode_sMeijerG(er e, slong prec)
{
//std::cout << "ncode_sMeijerG(" << prec << "): " << e << std::endl;
    if (!ehas_head_sym_length(e, gs.sym_sMeijerG.get(), 3))
        return ecopy(e);

    return ecopy(e);
}


// PFQEquation[a, b, x0, x_Symbol, theta_Symbol]
ex dcode_hypPFQEquation(er e)
{
//std::cout << "dcode_hypPFQEquation: " << e << std::endl;
    assert(ehas_head_sym(e, gs.sym_hypPFQEquation.get()));

    if (elength(e) != 5)
        return ecopy(e);

    rfmpq_t r;
    std::vector<xfmpq_t> a, b;

    if (!from_ex(a, echild(e,1)) || !from_ex(b, echild(e,2)))
        return ecopy(e);

    if (eis_zero(echild(e,3)))
    {
        LEquation<xfmpq_t> eq;
        pfq_equ_at_zero<rfmpq_t>(r, eq, a, b);
        complete_xoutside<rfmpq_t>(r, eq.data);
        return emake_list(to_ex(eq.data.xoutside, echild(e,5), echild(e,4)));
    }
    else
    {
        LEquation<xfmpq_t> eq;
        pfq_equ_at_zero<rfmpq_t>(r, eq, a, b);
        complete_xoutside<rfmpq_t>(r, eq.data);
        uex res1(to_ex(eq.data.xoutside, echild(e,5), echild(e,4)));

        REquation<xfmpq_t> eq2;
        pfq_equ_at_infinity<rfmpq_t>(r, eq, a, b);
        convert<rfmpq_t>(r, eq, eq2);
        complete_xoutside<rfmpq_t>(r, eq.data);
        uex res2(to_ex(eq.data.xoutside, echild(e,5), echild(e,4)));

        return emake_list(res1.release(), res2.release());
    }
}
